const int mod = 1e9 + 7;
const int N = 2e6 + 5;

bool vis[N];
ll fac[N],Finv[N],f[N];
int p[N],mu[N],cnt;
struct
{
    // è´¨æ•°ï¼Œæ¬§æ‹‰å‡½æ•°ï¼Œè«æ¯”ä¹Œæ–¯å‡½æ•°

        mu[1] = 1;
        cnt = 0;
        FOR(i, 2, N - 1)
        {
            if(!vis[i]) p[++cnt] = i, mu[i] = -1;
            for(ll j=1;j<=cnt && (ll)i * p[j] <N; j++)
            {
                vis[i*p[j]] = 1;
                if(!(i%p[j])) break;
                mu[i*p[j]] = -mu[i];
            }
        }
        f[1]= 1;
        FOR(i,2,N-1) f[i] = (f[i-1] + f[i-2]) % (mod-1);

        fac[0] = 1ll;
        FOR(i,1,N-1) fac[i]=fac[i-1]*i%mod;
        Finv[N-1] = fpow(fac[N-1],mod-2,mod);
        _FOR(i,N-2,0) Finv[i]=Finv[i+1]*(i+1)%mod;
    }
    ll C(ll n,ll m)
    {
        if(n<0||m<0||m>n) return 0;
        if(n  == m || m == 0) return 1;
        return fac[n] * Finv[m] %mod * Finv[n-m] %mod;
    }
}O_O;

        {
