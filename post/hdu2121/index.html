「HDU 2121」Ice_cream’s world II

2019-07-03 #最小树形图

After awarded lands to ACMers, the queen want to choose a city be her capital. This is an important event in ice_cream world, and it also a very difficult problem, because the world have N cities and M roads, every road was directed. Wiskey is a chief engineer in ice_cream world. The queen asked Wiskey must find a suitable location to establish the capital, beautify the roads which let capital can visit each city and the project’s cost as less as better. If Wiskey can’t fulfill the queen’s require, he will be punishing.

题解

不定点最小树形图
建立一个超级原点，超级原点对其他边连一条极大权值的边，然后以这个点为起点跑朱刘算法
如果存在树形图，那么只会用到一条超级原点的边，否则无解
找原点通过汇点连接的边来找，将原图的边和后面加的边区分开来，使用后面加边连接的点一定是起始点

代码

#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 2e4+10;
struct
{
    int x,y,z;
}e[maxd];
int vis[1005],pre[1005],in[1005],id[1005];
int n,m,inf,ans,pos,cnt,tot;
int zlsf()
{
    int root = n-1; ans = 0, pos =0;
    while(1)
    {
        fill(in,in+1+n,2e9);
        fill(id,id+1+n,-1);
        fill(vis,vis+1+n,-1);

        for(int i=0;i<tot;i++)
        {
            int x = e[i].x, y =e[i].y;
            //printf("%d %d %d\n",x,y,e[i].z);
            if(e[i].z < in[y] && x!=y)
            {
                pre[y] = x , in[y] = e[i].z; 
                if(x == root) pos = i;
            }
        }
        // for(int i=0;i<n;i++) printf("%d %d\n",i,in[i]);
        // putchar(10);

        for(int i=0;i<n;i++)
            if(i!=root && in[i] == 2e9) return 0;

        in[root] = 0 ,cnt = 0;
        
        for(int i=0;i<n;i++)
        {
            ans += in[i]; 
            int v = i;
            while(vis[v] !=i && id[v] ==-1&& v!= root) vis[v] = i, v = pre[v];
            if(v!=root && id[v] == -1)
            {
                for(int u=pre[v]; u!=v ;u = pre[u]) id[u] = cnt;
                id[v] = cnt++;
            }
        }
        
        if(cnt == 0 ) return 1;
        for(int i=0;i<n;i++) if(id[i] == -1) id[i] = cnt++;
        for(int i=0;i<=tot;i++)
        {
            int x = e[i].x, y = e[i].y;
            e[i].x = id[x];
            e[i].y = id[y];
            if(id[x] != id[y]) e[i].z -= in[y];
        } 
        n = cnt;
        root = id[root];
    }
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    while(scanf("%d %d",&n,&m)!=EOF)
    {
        inf = 1;
        for(int i=0;i<m;i++)
        {
            scanf("%d %d %d",&e[i].x,&e[i].y,&e[i].z);
            inf += e[i].z;  
        }
        for(int i=0;i<n;i++) e[m+i].x = n, e[m+i].y = i , e[m+i].z = inf;
        tot = n+m, n++;
        if(!zlsf()||ans >= 2*inf)  printf("impossible\n\n");
        else printf("%d %d\n\n",ans-inf,pos-m);
        //printf("%d %d\n",ans,inf);
    }
    return 0;
}
