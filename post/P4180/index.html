「BJWC2010」 严格次小生成树

2019-05-05 LCA

小C最近学了很多最小生成树的算法，Prim算法、Kurskal算法、消圈算法等等。正当小C洋洋得意之时，小P又来泼小C冷水了。小P说，让小C求出一个无向图的次小生成树，而且这个次小生成树还得是严格次小的，也就是说：如果最小生成树选择的边集是EM，严格次小生成树选择的边集是ES，那么需要满足：(value(e)表示边e的权值)
sume∈EMvalue(e)<∑e∈ESvalue(e)sum_{e \in E_M} value(e) < \sum_{e \in E_S } value(e)sume∈EM​​value(e)<∑e∈ES​​value(e)

这下小 C 蒙了，他找到了你，希望你帮他解决这个问题。

[BJWC2010]严格次小生成树

题解

严格次小生成树，与最小生成树只有一条边不同
可以以此枚举出非生成树上的边，树加入边后必定形成环，将环上小于加入边的最大边删除，可以得到一个增量，次小生成树的权值就为最小生成树的权值加最小增量
求环上最短边可以用LCA倍增求解

代码

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
using namespace std;
const int maxd = 1e5+100;
struct node
{
    int x,y,flag;
    long long w;
    bool operator < (const node &other) const
    {
        return w < other.w;
    }
}arr[maxd<<2];
inline int read()
{
    int f=1,x=0;char ch;
    do{ch=getchar();if(ch=='-')f=-1;}while(ch<'0'||ch>'9');
    do{x=x*10+ch-'0';ch=getchar();}while(ch>='0'&&ch<='9');
    return f*x;
}
struct
{
    int v,net;
    long long w;
}e[maxd<<1];
int f[maxd],eid;
void insert(int a,int b,long long c)
{
    e[eid].net= f[a];
    e[eid].v= b;
    e[eid].w = c;
    f[a] = eid++;
}
int p[maxd][30],d[maxd];
long long  mx[maxd][20],mi[maxd][30];
int fa[maxd];
int n,m,a,b,c,x;
long long ans =1e14,sum=0ll;
void init()
{
    memset(f,-1,sizeof(f));
    memset(d,-1,sizeof(f));
    for(int i=1;i<=n;i++) fa[i] = i;
    eid = 0; d[1] = 0;
}
int gf(int x)
{
    if(fa[x] == x) return x;
    return fa[x] = gf(fa[x]);
}
void dfs(int x)
{
    for(int i = f[x]; i+1; i = e[i].net)
    {
        int v = e[i].v;
        if(d[v]==-1)
        {
            d[v] = d[x]+1;
            p[v][0] = x;
            mx[v][0] = e[i].w;
            mi[v][0] = -1e14;
            dfs(v);
        }    
    }
}
int lca(int a,int b,long long c)
{
    long long x=-1e14,y=-1e14;
    if(d[b] > d[a]) swap(a,b);
    for(int i = 15;i>=0;i--)
        if(d[p[a][i]]>=d[b])
        {
            long long k = 0;
            if(x!=mx[a][i]) k = min(x,mx[a][i]);
            x = max(x,mx[a][i]);
            y = max(max(y,mi[a][i]),k); 
            a = p[a][i];
        }
    if(a==b)
    {
        if(x == c) return y;
        return x;
    }
    for(int i = 15;i>=0;i--)
        if(p[a][i]!=p[b][i])
        {
            long long check[6];
            check[0] = x;check[1] = mx[b][i]; check[2]=mx[a][i];
            check[3] = y;check[4] = mi[b][i]; check[5]=mi[a][i];
            sort(check,check+6);
            x = check[5];
            for(int j=0;j<6;j++)
            {
                if(check[j] == x) break;
                y = check[j];
            }
            a=p[a][i],b = p[b][i];
        }
    long long check[6];
    check[0] = x;check[1] = mx[b][0]; check[2]=mx[a][0];
    check[3] = y;check[4] = mi[b][0]; check[5]=mi[a][0];
    sort(check,check+6);
    x = check[5];
    for(int i=0;i<6;i++)
    {
        if(check[i] == x) break;
        y = check[i];
    }
    if(x==c) return y;
    return x;
}
void solve()        //求最小生成树
{
    sort(arr+1,arr+m+1);
    for(int i=1;i<=m;i++)
    {
        int x = gf(arr[i].x), y = gf(arr[i].y);
        if(x == y) continue;
        sum += arr[i].w;
        fa[x] = y; 
        arr[i].flag = 1;
    }
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    n =read(), m =read();
    init();
    for(int i=1;i<=m;i++)       //读取所有边
        arr[i].x = read(), arr[i].y =read() , arr[i].w= (long long)read();
    solve();
    for(int i=1;i<=m;++i)   //  建树
    {
        if(arr[i].flag)
        {
            insert(arr[i].x,arr[i].y,arr[i].w);
            insert(arr[i].y,arr[i].x,arr[i].w);
        }
    }
    dfs(1);
    for(int j=1;j<=15;j++)
        for(int i=1;i<=n;i++){
            p[i][j] = p[p[i][j-1]][j-1];    // 父节点
            long long k = -1e14;
            if(mx[i][j-1]!=mx[p[i][j-1]][j-1])  // 如果两个点的最长边不同
            k = min(mx[i][j-1],mx[p[i][j-1]][j-1]); //边权小的那条可能成为次短边
            mx[i][j] = max(mx[i][j-1],mx[p[i][j-1]][j-1]);
            mi[i][j] = max(k,max(mi[i][j-1],mi[p[i][j-1]][j-1]));
        }
    for(int i=1;i<=m;i++)
        if(!arr[i].flag)
        {
            long long k = arr[i].w - lca(arr[i].x,arr[i].y,arr[i].w);
            ans = min(ans,k);   //  最小增量
        }
    printf("%lld",sum + ans);
    return 0;
}
