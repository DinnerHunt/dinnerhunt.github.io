题解

题面很有意思的一道DP
将大于2000的值视为2001，并不会影响最后答案的结果

代码

#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <string>
using namespace std;
const int maxd = 105;
int n,m,k;
long long dp[maxd][maxd*20],a[maxd*20],b[maxd],c[maxd];  // 第i天后，保留j节点的最大节点数
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    int n,m,k;
    scanf("%d %d %d",&n,&m,&k);
    for(int i=0;i<=k;i++) scanf("%lld",&a[i]);
    for(int i=1;i<=m;i++) scanf("%lld %lld",&b[i],&c[i]);
    memset(dp,-1,sizeof(dp));
    dp[1][0] = 0;
    for(int i=1;i<=n+1;i++)
    {
        for(int j=2001;j>=0;j--)
        {
            if(dp[i][j] == -1) continue;
            for(int l=1;l<=m;l++)
            {
                if(j >= b[l])
                dp[i][j-b[l]] = max(dp[i][j-b[l]],dp[i][j] + c[l]);
            }
        }
        for(int j=0;j<=2001;j++)
        {
            if(dp[i][j] == -1) continue;
            if(j+a[j] > 2000) dp[i+1][2001] = max(dp[i+1][2001],dp[i][j]+j+a[j]-2000);
            else dp[i+1][j+a[j]] = max(dp[i+1][j+a[j]],dp[i][j]);
        }
    }
    long long ans = 0ll;
    // for(int i=1;i<=n+1;i++){
    //     for(int j=0;j<=k+1;j++)
    //         printf("%lld ",dp[i][j]);
    //     putchar(10);
    // }
    for(int i=0;i<=2000;i++)
    {
        if(dp[n+1][i] == -1) continue;
        ans = max(dp[n+1][i]+i,ans);
    }
    if(dp[n+1][2001]) max(ans,dp[n+1][2000+1]+2000);
    printf("%lld",ans);
    return 0;
}
