  • 组合数
struct
{
    ll fac[N],inv[N],Finv[N];
    void init()
    {
        fac[0] = 1ll, Finv[0]=1ll,inv[1]=1ll;
        for(int i=2;i<N;i++) inv[i]=(mod-mod/i)*inv[mod%i]%mod;
        for(int i=1;i<N;i++)
        {
            fac[i]=fac[i-1]*i%mod;
            Finv[i]=Finv[i-1]*inv[i]%mod;
        }
    }
    ll C(ll n,ll m)
    {
        if(n<0||m<0||m>n) return 0;
        if(n == m || m == 0) return 1;
        return fac[n] * Finv[m] %mod * Finv[n-m] %mod;
    }
    ll A(ll n,ll m)
    {
        return fac[n] * Finv[n-m] % mod;
    }
}O_O;
// 当K比较小，求单次组合数时
ll comb(ll n,ll k) {
    if(n<k||k<0)return 0;
    k=min(k,n-k);
    if(k==0) return 1;
    ll ret=1;
    vector<ll> inv(k+1);
    inv[1]=1;
    for(int i=2;i<=k;++i) inv[i]=(mod-(mod/i))*inv[mod%i]%mod;
    for(int i=n-k+1;i<=n;++i) ret=ret*i%mod;
    for(int i=1;i<=k;++i) ret=ret*inv[i]%mod;
    return ret;
}
