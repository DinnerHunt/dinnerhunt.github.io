void init()
{
    for(int i=2;i<maxd;i++)
    {
        if(!vis[i])
        {
            pri[++cnt]=i;
            phi[i]=i-1;
        }
        for(int j=1;j<=cnt&&i*pri[j]<maxd;j++)
        {
            vis[i*pri[j]]=1;
            if(!(i%pri[j]))
            {
                phi[i*pri[j]]=phi[i]*pri[j];
                break;
            }
            else phi[i*pri[j]]=phi[i]*(pri[j]-1);
        }
    }
}

void init()
{
    mu[1] = 1;
    for(int i=2;i<maxd;i++)
    {
        if(!vis[i]) prime[++cnt] = i, mu[i] = -1;
        for(int j=1;j<=cnt && i*prime[j] <= maxd;j++)
        {
            vis[i*prime[j]] = 1;
            if(i%prime[j] == 0) break;
            mu[i*prime[j]] = -mu[i]; 
        }
    }
}
