「AtCoder 133E」 Virus Tree 2

2019-09-29 DFS 组合数学

You are given a tree with N vertices and N−1 edges. The vertices are numbered 1 to N, and the i-th edge connects Vertex ai and bi.

You have coloring materials of K colors. For each vertex in the tree, you will choose one of the K colors to paint it, so that the following condition is satisfied:

If the distance between two different vertices x and y is less than or equal to two, x and y have different colors.
How many ways are there to paint the tree? Find the count modulo 1 000 000 007.

题解

在DFS中考虑每个节点的可选择方案数，最后统计即可

代码

#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
const int maxd = 1e5+10;
int t[maxd],flag,v[maxd];
vector<int> g[maxd];
int n,m;
const int mod =1e9+7;
void dfs(int x,int fa = 0,int d = 1)
{
    int now = m-1; 
    if(d > 2) v[x]--;
    if(v[x]<=0) 
    {
        flag=1;
        return;
    }
    //printf("%d %d\n",x,v[x]);
    for(int i=0;i<g[x].size();i++)
    {
        int k = g[x][i];
        if(k==fa) continue;
        v[k] = now;
        //printf("    %d %d\n",k,v[x]);
        dfs(k,x,d+1);
        now--;
    }
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%d %d",&n,&m);
    for(int i=1;i<n;i++)
    {
        int a,b;
        scanf("%d %d",&a,&b);
        g[a].push_back(b);
        g[b].push_back(a);
    }
    v[1] = m;dfs(1);
    long long ans = 1ll;
    // for(int i=1;i<=n;i++)
    // {
    //     printf("%d ",v[i]);
    // }
    if(flag) printf("0\n");
    else 
    {
        for(int i=1;i<=n;i++)
            ans = 1ll*ans * v[i] %mod;
        printf("%lld\n",ans);
    }
    return 0;
}
