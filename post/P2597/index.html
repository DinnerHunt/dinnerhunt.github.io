жҲ‘д»¬зҺ°еңЁд»Һдё“дёҡдёҖзӮ№зҡ„и§’еәҰжқҘзңӢиҝҷдёӘй—®йўҳгҖӮжҲ‘д»¬з”ЁдёҖз§ҚеҸ«еҒҡйЈҹзү©зҪ‘зҡ„жңүеҗ‘еӣҫжқҘжҸҸиҝ°з”ҹзү©д№Ӣй—ҙзҡ„е…ізі»пјҡ

еӣҫдёӯжңүдёҖдәӣзӮ№жІЎжңүиҝһеҮәиҫ№пјҢиҝҷдәӣзӮ№д»ЈиЎЁзҡ„з”ҹзү©йғҪжҳҜз”ҹдә§иҖ…пјҢеҸҜд»ҘйҖҡиҝҮе…үеҗҲдҪңз”ЁжқҘз”ҹеӯҳпјӣ иҖҢжңүиҝһеҮәиҫ№зҡ„зӮ№д»ЈиЎЁзҡ„йғҪжҳҜж¶Ҳиҙ№иҖ…пјҢе®ғд»¬еҝ…йЎ»йҖҡиҝҮеҗғе…¶д»–з”ҹзү©жқҘз”ҹеӯҳгҖӮ

еҰӮжһңжҹҗдёӘж¶Ҳиҙ№иҖ…зҡ„жүҖжңүйЈҹзү©йғҪзҒӯз»қдәҶпјҢе®ғдјҡи·ҹзқҖзҒӯз»қгҖӮ

жҲ‘д»¬е®ҡд№үдёҖдёӘз”ҹзү©еңЁйЈҹзү©зҪ‘дёӯзҡ„вҖңзҒҫйҡҫеҖјвҖқдёәпјҢеҰӮжһңе®ғзӘҒз„¶зҒӯз»қпјҢйӮЈд№Ҳдјҡи·ҹзқҖдёҖиө·зҒӯз»қзҡ„з”ҹзү©зҡ„з§Қж•°гҖӮ

дёҫдёӘдҫӢеӯҗпјҡеңЁдёҖдёӘиҚүеңәдёҠпјҢз”ҹзү©д№Ӣй—ҙзҡ„е…ізі»жҳҜпјҡ

#include <cstdio>
#include <queue>
#include <vector>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 1e5+10;
int fa[maxd][16],degree[maxd],p[maxd],cnt,d[maxd];
int n,x,y,ans[maxd];
vector<int> G[maxd],food[maxd],son[maxd];
//G еҺҹеӣҫпјҢG1[i] дёәiиҠӮзӮ№зҡ„зҢҺзү©
void topsort()
{
    queue<int> que;
    for(int i=1;i<=n;i++)
        if(!degree[i]) G[0].push_back(i),food[i].push_back(0),degree[i]++;
    //еҗҢдёҖеӣҫдёҠеӯҳеңЁдёӨдёӘе…ҘеәҰдёә0зҡ„зӮ№пјҢеҲҷдјҡжүҫдёҚеҲ°lca
    que.push(0);
    while(!que.empty())
    {
        x = que.front();p[++cnt] = x;que.pop();
        for(int i=0;i<G[x].size();i++)
        {
            int v =G[x][i];
            --degree[v];
            if(!degree[v])que.push(v);
        }
    }
}
int lca(int x,int y)
{
    if(d[x] < d[y]) swap(x,y);
    for(int i=15;i>=0;i--) if(d[fa[x][i]] >= d[y]) x = fa[x][i];
    if(x==y) return x;
    for(int i=15;i>=0;i--) if(fa[x][i] != fa[y][i]) x = fa[x][i],y = fa[y][i];
    return fa[x][0];
}
int dfs(int x)
{
    ans[x] =1;
    for(int i=0;i<son[x].size();i++) ans[x] += dfs(son[x][i]);
    return ans[x];
}
int main()
{
    // freopen("a.in","r",stdin);

    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        while(scanf("%d",&x)!=EOF&&x)
        {
            G[x].push_back(i);
            food[i].push_back(x); 
            ++degree[i];
        }
    }
    topsort();
    //printf("%d ",cnt);
    for(int i=2;i<=cnt;i++)
    {
        x = p[i];y = food[x][0];
        //printf("%d ",x);
        for(int j=1;j< food[x].size();j++) y = lca(y,food[x][j]);
        son[y].push_back(x); //зӮ№yжҳҜзӮ№xзҡ„ж”Ҝй…ҚзӮ№
        d[x] = d[y]+1;
        fa[x][0] = y;
        for(int j=1;j<=15;j++) fa[x][j] = fa[fa[x][j-1]][j-1]; //еҖҚеўһ
    }
    dfs(0);
    for(int i=1;i<=n;i++) printf("%d\n",ans[i]-1);
    return 0;
}
