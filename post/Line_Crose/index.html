int is(ln a,ln b)
{
    if( min(a.s.x,a.e.x) <= max(b.s.x,b.e.x) &&
        max(a.s.x,a.e.x) >= min(b.s.x,b.e.x) &&
        min(a.s.y,a.e.y) <= max(b.s.y,b.e.y) &&
        max(a.s.y,a.e.y) >= min(b.s.y,b.e.y) &&
        1ll * mul(b.s,a.s,b.e)*mul(b.s,b.e,a.e) >= 0 &&
        1ll * mul(a.s,b.s,a.e)*mul(a.s,a.e,b.e) >= 0 )
        return 1;
    else return 0;
}

double min(double a, double b)
{
    return a < b ? a : b;
}

double max(double a, double b)
{
    return a > b ? a : b;
}
bool inter(pt a, pt b, pt c, pt d)
{
    if (min(a.x, b.x) > max(c.x, d.x) || min(a.y, b.y) > max(c.y, d.y) || min(c.x, d.x) > max(a.x, b.x) || min(c.y, d.y) > max(a.y, b.y))
    {
        return 0;
    }
    double h, i, j, k;
    h = (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
    i = (b.x - a.x) * (d.y - a.y) - (b.y - a.y) * (d.x - a.x);
    j = (d.x - c.x) * (a.y - c.y) - (d.y - c.y) * (a.x - c.x);
    k = (d.x - c.x) * (b.y - c.y) - (d.y - c.y) * (b.x - c.x);
    return h * i <= eps && j * k <= eps;
}
