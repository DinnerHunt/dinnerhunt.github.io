#include <cstdio>
#include <queue>
#include <algorithm>
#include <cstring>
using namespace std;
const int maxd = 2e6+10;
char str[55];
bool vis[1005][1<<12];
int n,t,last[maxd],ans[maxd];
struct node
{
    int id,sta,k;
};
struct 
{
    int ch[maxd][26],fail[maxd],cnt[maxd],ans[maxd],tot;
    void insert(char *str,int k)
    {
        int p = 0;
        for(int i=0;str[i];i++)
        {
            if(!ch[p][str[i]-'A']) ch[p][str[i]-'A'] = ++tot;
            p = ch[p][str[i]-'A'];
            ans[p]++;
        }
        cnt[p] |= 1<<k>>1;
        //printf("%d %d\n",cnt[p],p);
    }
    void build()
    {
        memset(fail,0,sizeof(fail));
        queue<int> que;
        for(int i=0;i<26;i++) if(ch[0][i])
        {
            que.push(ch[0][i]);
        }
        while(!que.empty())
        {
            int p = que.front();que.pop();
            for(int i=0;i<26;i++)
            {
                if(ch[p][i])
                {
                    fail[ch[p][i]] = ch[fail[p]][i];
                    que.push(ch[p][i]);
                }
                else ch[p][i] = ch[fail[p]][i];
            }
            int i;
            for(i=fail[p];i&&!cnt[i]; i= fail[i]);
            cnt[p] |= cnt[i];
        }
    }
    void print(int x)
    {
        if(!x) return;
        print(last[x]);
        printf("%c",ans[x]+'A');
    }
    void solve()
    {
        queue<node> que;
        node a; a.id = 0, a.sta =0,a.k=0;
        que.push(a);
        vis[0][0] = 1;
        int id,sta,k,now;
        while(!que.empty())
        {
            node nod = que.front(); que.pop();
            id = nod.id;
            k = nod.k;
            sta = nod.sta;
            if(sta == (1<<n)-1)
            {
                print(k);
                return;
            }
            for(int i=0;i<26;i++)
            {
                now = sta|cnt[ch[id][i]];
                //printf("%d %d %d\n",id,i,ch[id][i]);
                if(vis[ch[id][i]][now]) continue;
                vis[ch[id][i]][now] = 1;
                last[++t] = k, ans[t] = i;
                a.id = ch[id][i],a.sta = now,a.k = t;
                //printf("%d %d %d %d %d\n",id,i,a.id,a.sta,a.k);
                que.push(a);
            }
        }
    }
}AC;
int main()
{
    // freopen("a.in","r",stdin);

    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%s",str);
        AC.insert(str,i);
    }
    AC.build();
    AC.solve();
    return 0;
}
