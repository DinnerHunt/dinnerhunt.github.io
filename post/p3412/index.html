const int mod = 998244353;
const double eps = 1e-8;
const int N = 2e5+10;
V G[N];
ll sum,fa[N],sz[N],d[N];
void dfs(ll x,ll f)
{
    d[x] = G[x].size();
    sz[x] = 1; fa[x] = f;
    for(auto v: G[x])
    {
        if(v == f) continue;
        dfs(v,x); sz[x] += sz[v];
        d[x] += d[v];
    }
}

}
