构造字符串的HASH表，O(1) 时间内对其子串进行判断是否相等

#include <cstdio>
#include <algorithm>
using namespace std;
const int maxd = 1e5+10;
const int P = 13331;
unsigned long long pre[maxd],hash[maxd];
int len;
char str[maxd];
void init()
{
 	for(int i=1,pre[0]=1;i<maxd;i++) pre[i] = pre[i-1]*P;
}
void hashPro()
{
	for(int i=1;i<=len;i++) hash[i] = hash[i-1] * P + (str[i] - 'a' +1);
}
unsigned gethash(int l,int r)
{
	return hash[r] - hash[l-1] * pre[r-l+1];
}
int main()
{
	//freopen("a.in","r",stdin);
	//freopen("k.out","w",stdout);
	scanf("%s",str+1); len = strlen(str+1);
	init();hashPro();
	return 0;
}
