后缀自动机

2019-05-04

P3804 【模版】后缀自动机

#include <cstdio>
#include <map>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 2e6+10;

long long num[maxd],ans;
struct SAM
{
    struct state
    {
        int len ,link;
        int net[30];
       	state(){memset(net,0,sizeof(net));len = link = 0;}
    }st[maxd];
    int sz,last;
    SAM()
    {
        st[1].len = 0;
        st[1].link = 0;
        sz = 1;
        last = 1;
    }
    void add(char c)
    {
        int cur = ++sz,p=last;
        num[cur] = 1;
        st[cur].len = st[last].len+1;
        while(p&& !st[p].net[c])
        {
            st[p].net[c] = cur;
            p = st[p].link;
        }
        if(!p) st[cur].link = 1;
        else 
        {
            int q = st[p].net[c];	// 该节点的指向c的下一节点
            if(st[p].len + 1 == st[q].len) st[cur].link = q;
            else 
            {
                int clone = ++sz;
                st[clone].len = st[p].len + 1;
                memcpy(st[clone].net,st[q].net,sizeof(st[q].net));
                st[clone].link = st[q].link;
                while(p && st[p].net[c] == q)
                {
                    st[p].net[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
}sa;
struct G
{
    struct node
    {
        int v,net;
    }e[maxd];
    int f[maxd],eid;
    G()
    {
        memset(f,-1,sizeof(f));
        eid = 0;
    }
    void insert(int x,int y)
    {
        e[eid].v = y;
        e[eid].net =f[x];
        f[x] = eid++;
        //printf("%d %d\n",x,y);
    }
}mp;
void dfs(int x)
{
    for(int i=mp.f[x];i+1;i=mp.e[i].net)
    {
        int v = mp.e[i].v;
        dfs(v);
        num[x] += num[v];
    }
    if(num[x]!=1) ans = max(ans,num[x] * sa.st[x].len);
}
char str[maxd];
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%s",str);
    for(int i=0;str[i];i++) sa.add(str[i]-'a');
    //printf("%d ",sa.st[1].link);
    for(int i=2;i<=sa.sz;i++) mp.insert(sa.st[i].link,i);
    dfs(1);
    printf("%lld",ans);
    return 0;
}
