#include <cstring>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
const int maxn = 205;
 
int n, k;
vector<int> G[maxn];
int dp[maxn][maxn], a[maxn];
 
void dfs(int x, int f = 0)
{
    dp[x][0] = a[x];
    for(int v:G[x])
    {
        if(v == f) continue;
        dfs(v, x);
        dp[x][0] += dp[v][k];
    }
    for(int i = 1; i < n; i++)
    {
        for(auto it : G[x])
        {
            if(it == f) continue;
            int cnt = dp[it][i-1];
            for(auto other:G[x])
            {
                if(other == it || other == f) continue;
                cnt += dp[other][max(i-1,k-i)];
            }
            dp[x][i] = max(dp[x][i],cnt);
        }
    }
    for(int i = n-1; i >= 0; i--)
        dp[x][i-1] = max(dp[x][i-1], dp[x][i]);
}
