#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 24000010;
int fail[maxd],len;
char s[maxd],str[maxd];
void init()
{
	str[0] = str[1] = '#';
	len = strlen(s);
	for(int i=0;i<len;i++)
	{
		str[i*2+2] = s[i];
		str[i*2+3] = '#';  
	}
	len = len*2+2;
}
void manacher()
{
	int match = 0, pos=0,ans = 0;
	for(int i=1;i<len;i++)
	{
		if(i<match) fail[i] = min(fail[(pos<<1)-i],match-i);
		else fail[i] = 1;
		while(str[fail[i]+i] == str[i-fail[i]]) fail[i]++;
		if(i+fail[i]>match)
		{
			match = fail[i]+i;
			pos = i;
		}
		ans = max(fail[i],ans);
	}
	printf("%d",ans-1);
}
int main()
{
	// freopen("a.in","r",stdin);
	// freopen("k.out","w",stdout);
	scanf("%s",s);
	init();
	manacher();
	return 0;
}
