You have a password which you often type вҖ” a string рқ‘  of length рқ‘ӣ. Every character of this string is one of the first рқ‘ҡ lowercase Latin letters.

Since you spend a lot of time typing it, you want to buy a new keyboard.

A keyboard is a permutation of the first рқ‘ҡ Latin letters. For example, if рқ‘ҡ=3, then there are six possible keyboards: abc, acb, bac, bca, cab and cba.

Since you type your password with one finger, you need to spend time moving your finger from one password character to the next. The time to move from character рқ‘ рқ‘– to character рқ‘ рқ‘–+1 is equal to the distance between these characters on keyboard. The total time you have to spend typing the password with a keyboard is called the slowness of this keyboard.

#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 1e5+10;
char s[maxd];
int a[maxd],cnt[25][25],dp[1<<20];
int n,m;
int main()
{
	// freopen("a.in","r",stdin);
	// freopen("k.out","w",stdout);
	scanf("%d %d %s",&n,&m,s+1); 
	int k = m;m=1<<m;
	for(int i=2;i<=n;i++)
	{
		int a = s[i-1] - 'a';
		int b = s[i] - 'a';
		if(a!=b)
		{
			cnt[a][b]++;
			cnt[b][a]++;
		}
	}
	dp[0]=0;
	for(int i=1;i<m;i++)
	{
		dp[i] = 1e9;
		for(int j=0;j<k;j++)
			if(((i>>j)&1))  dp[i] = min(dp[i^(1<<j)],dp[i]);

			if((i>>j)&1)
				for(int l=0;l<k;l++)
					if(!((i>>l)&1)) dp[i] += cnt[j][l];
	}
	printf("%d",dp[m-1]);
	return 0;
}
