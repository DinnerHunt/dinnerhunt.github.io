「FFT」多项式乘法卷积

2019-08-29 FFT

P3803 多项式乘法（FFT）

给定两个多项式F(x), G(x), 求它们的卷积

#include <cstdio>
#include <cmath>
#include <algorithm>
#include <cstring>
using namespace std;
const double Pi = acos(-1.0);
const int maxd = 5e6+10;
struct complex
{
    double x,y;
    complex(double _x=0,double _y=0){x = _x , y = _y;}
}a[maxd],b[maxd];
complex operator + (complex a,complex b) { return complex(a.x+b.x,a.y+b.y);}
complex operator - (complex a,complex b) { return complex(a.x-b.x,a.y-b.y);}
complex operator * (complex a,complex b) { return complex(a.x*b.x-a.y*b.y,a.x*b.y+a.y*b.x);}
int n,m,len,pos[maxd],limit=1;
void fft(complex *A, int symbol)
{
    for(int i=0;i<limit;i++) if(i<pos[i]) swap(A[i],A[pos[i]]);
    for(int mid = 1; mid <limit;mid<<=1)
    {
        complex unity(cos(Pi/mid),symbol * sin(Pi/mid));
        for(int R = mid <<1, j = 0; j < limit; j+=R)
        {
            complex w(1,0);
            for(int k=0;k<mid;k++, w = w *unity)
            {
                complex x = A[j+k], y = w * A[j+mid+k];
                A[j+k] = x+y;
                A[j+mid+k] = x-y;
            }
        }
    }
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    int n,m; scanf("%d %d",&n,&m);
    for(int i=0;i<=n;i++) scanf("%lf",&a[i].x);
    for(int i=0;i<=m;i++) scanf("%lf",&b[i].x);
    while(limit<=n+m) limit<<=1,len++;
    for(int i=0;i<limit;i++) pos[i] = (pos[i>>1] >> 1) | ((i&1) << (len-1));
    fft(a,1);fft(b,1);
    for(int i=0;i<=limit;i++) a[i] = a[i]*b[i];
    fft(a,-1);
    for(int i=0;i<=n+m;i++) printf("%d ",(int)((a[i].x)/limit+0.5));

    return 0;
}
