    for(int i=0;i<limit;i++) if(i<pos[i]) swap(A[i],A[pos[i]]);
    for(int mid = 1; mid <limit;mid<<=1)

        for(int R = mid <<1, j = 0; j < limit; j+=R)

                A[j+k] = x+y;
                A[j+mid+k] = x-y;
            }
        }
    }
}

        }

    // dft(x,x,1), idft(x,x,-1)
    void dft(ll *a, int n, int f) {
        FOR(i, 0, n-1) {
            r[i] = r[i / 2] / 2 + (i & 1) * (n / 2);
            if(i < r[i]) swap(a[i], a[r[i]]);
        } ll b;
        for(int i = 1; i < n; i *= 2) for(int j = 0; j < n; j += 2 * i) FOR(k, 0, i-1)
        b = a[i + j + k] * w[i + k], a[i + j + k] = (a[j + k] - b) % mod, a[j + k] = (a[j + k] + b) % mod;
        if(f == -1) {
            reverse(a + 1, a + n);
            b = fpow(n, mod - 2,mod);
            FOR(i, 0, n-1) a[i] = (a[i] + mod) * b % mod;
        }
    }
