#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
using namespace std;
const int maxd = 1e6+10;
const int maxn = 1e4+10;
struct
{
    int v,net;
}e[maxd<<2];
int f[maxd<<1],eid;
void insert(int x,int y)
{
    e[eid].v = y;
    e[eid].net = f[x];
    f[x] = eid++;
}
bool vis[maxd<<1];
int ans[maxd<<1],n,m;

void init()
{
    memset(f,-1,sizeof(f));
    eid=0;
}
bool dfs(int x)
{
    for(int i=f[x];i+1;i=e[i].net)
    {
        int v = e[i].v;
        if(!vis[v])
        {
            vis[v] = true;
            if(ans[v] ==-1||dfs(ans[v]))
            {
                ans[v] = x;
                return true; 
            }
        }
    }
    return false;
}
int maxmatch()
{
    int cnt = 0;
    memset(ans,-1,sizeof(ans));
    for(int i=1;i<maxd;i++)
    {
        memset(vis,0,sizeof(vis));
        if(!dfs(i)) return cnt;
        cnt++;
    }
    return cnt;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%d",&n);init();
    for(int i=1;i<=n;i++)
    {
        int x,y;
        scanf("%d %d",&x,&y);
        insert(x,i+maxn); insert(i+maxn,x);
        insert(i+maxn,y); insert(y,i+maxn);
    }
    printf("%d",maxmatch());
    return 0;
}
