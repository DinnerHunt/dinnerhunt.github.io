иҝҷж ·дёәд»Җд№ҲжҳҜжӯЈзЎ®зҡ„е‘ўпјҹжҲ‘д»¬еҸҜд»ҘиҖғиҷ‘еҢҲзүҷеҲ©з®—жі•зҡ„е…·дҪ“иҝҮзЁӢпјҡеңЁеҢ№й…ҚеҖјдёә i зҡ„жҠҖиғҪж—¶пјҢйӮЈд№Ҳ 1 еҲ° iвҲ’1 зҡ„еұһжҖ§иӮҜе®ҡе·Із»ҸеҢ№й…Қе®ҢжҲҗпјҢжүҖд»ҘеҰӮжһң i еҜ№еә”зҡ„зј–еҸ· j иў«еҢ№й…ҚдәҶзҡ„иҜқпјҢйӮЈд№Ҳе°ұи®©еҢ№й…Қ j зҡ„йӮЈдёӘеұһжҖ§ p еҶҚеҺ»жүҫеҲ«зҡ„зү©е“Ғж ҮеҸ·еҢ№й…ҚпјҢеҪўиұЎең°иҜҙпјҢе°ұжҳҜз”ЁеҲ«зҡ„зү©е“ҒжқҘйҮҠж”ҫж”»еҮ»еҠӣдёә p зҡ„иҝҷдёӘжҠҖиғҪпјҢз”Ё j иҝҷдёӘзү©е“ҒйҮҠж”ҫж”»еҮ»еҠӣдёә i зҡ„жҠҖиғҪгҖӮеҰӮжһңжүҫеҲ°иҝҷж ·дёҖжқЎеўһе№ҝи·ҜпјҢйӮЈд№Ҳе°ұиҜҙжҳҺеҪ“еүҚеҸҜд»ҘеҢ№й…ҚпјҢans++гҖӮ

#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
using namespace std;
const int maxd = 1e6+10;
const int maxn = 1e4+10;
struct
{
    int v,net;
}e[maxd<<2];
int f[maxd<<1],eid;
void insert(int x,int y)
{
    e[eid].v = y;
    e[eid].net = f[x];
    f[x] = eid++;
}
bool vis[maxd<<1];
int ans[maxd<<1],n,m;

void init()
{
    memset(f,-1,sizeof(f));
    eid=0;
}
bool dfs(int x)
{
    for(int i=f[x];i+1;i=e[i].net)
    {
        int v = e[i].v;
        if(!vis[v])
        {
            vis[v] = true;
            if(ans[v] ==-1||dfs(ans[v]))
            {
                ans[v] = x;
                return true; 
            }
        }
    }
    return false;
}
int maxmatch()
{
    int cnt = 0;
    memset(ans,-1,sizeof(ans));
    for(int i=1;i<maxd;i++)
    {
        memset(vis,0,sizeof(vis));
        if(!dfs(i)) return cnt;
        cnt++;
    }
    return cnt;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%d",&n);init();
    for(int i=1;i<=n;i++)
    {
        int x,y;
        scanf("%d %d",&x,&y);
        insert(x,i+maxn); insert(i+maxn,x);
        insert(i+maxn,y); insert(y,i+maxn);
    }
    printf("%d",maxmatch());
    return 0;
}
