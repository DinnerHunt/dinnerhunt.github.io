P3806 点分治

给定一棵有n个点的树

询问树上距离为k的点对是否存在。

#include <cstdio>
#include <bitset>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 1e5;
struct node
{
	int v,net,dis;
}e[maxd*2];
int f[maxd],eid;
int size[maxd],son[maxd],d[maxd],vis[maxd],tot;
int n,m,root,cnt,flag,ans[10000010];
void init()
{
	memset(f,-1,sizeof(f));
	eid = 0;
}
void insert(int a,int b,int dis)
{
	e[eid].v = b;
	e[eid].net = f[a];
	e[eid].dis = dis;
	f[a] = eid++;
}

void Get_Root(int x,int fa) //找根
{
	son[x] = 0,size[x] = 1;
	for(int i=f[x];i+1;i=e[i].net)
	{
		int v = e[i].v;
		if(v==fa||vis[v]) continue;
		Get_Root(v,x);
		son[x] = max(son[x],size[v]);
		size[x] += size[v];		
	}
	son[x] = max(son[x],tot - size[x]);
	if(son[x] < son[root]) root = x;
}
void Get_dis(int x,int fa,int len) // 统计当前重心到各个点的距离
{
	for(int i=f[x];i+1;i=e[i].net)
	{
		int v = e[i].v;
		if(vis[v]||v == fa) continue;
		d[++cnt] = len + e[i].dis;
		Get_dis(v,x,d[cnt]);
	}
}
void Get_Ans(int x,int dis,int w)	//统计当前点的符合条件数
{
	d[cnt=1] = dis;	//	加入0，覆盖一条边的统计
	Get_dis(x,0,dis);
	int l = 1,k = 0;
	for(int i=1;i<=cnt;i++)
		for(int j=1;j<=cnt;j++) 
			if(i!=j) ans[d[i]+d[j]] += w;
}
void Point_Divid(int x)
{
	vis[x] = 1;
	Get_Ans(x,0,1); 
	for(int i=f[x];i+1;i=e[i].net)
	{
		int v = e[i].v;
		if(vis[v]) continue;
		Get_Ans(v,e[i].dis,-1);
		tot = size[v];root = 0;
		Get_Root(v,x),Point_Divid(root);
	}
}

int main()
{
	// freopen("a.in","r",stdin);
	// freopen("k.out","w",stdout);
	init(); scanf("%d %d",&n,&m);
	for(int i=1;i<n;i++)
	{
		int x,y,z;
		scanf("%d %d %d",&x,&y,&z);
		insert(x,y,z);
		insert(y,x,z);
	}
	tot = n; son[0] = 1e9;Get_Root(1,0);
	Point_Divid(root);
	for(int i=1;i<=m;i++)
	{
		int x; scanf("%d",&x);
		if(ans[x])printf("AYE\n");
		else printf("NAY\n");
	}
	return 0;
}
