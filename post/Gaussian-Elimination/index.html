P3389 高斯消元

给定一个线性方程组，对其求解

#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
const double eps = 1e-8;
const int maxd = 100+5;
double arr[maxd][maxd],ans[maxd];
int n;
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=n+1;j++)
            scanf("%lf",&arr[i][j]);
    for(int i=1;i<=n;i++)
    {
        int p = i;
        for(int j=i;j<=n;j++)   //找系数最大的点
            if(fabs(arr[j][i]) < fabs(arr[p][i])) p=j;

        if(fabs(arr[i][i])<=eps) 
        {
            printf("No Solution\n");
            return 0;
        }
        if(i!=p) swap(arr[i],arr[p]);
        double div = arr[i][i];
        for(int j=i+1;j<=n+1;j++) arr[i][j] /= div; //列中所有系数全部除以该数
        for(int j=i+1;j<=n;j++)
        {
            div = arr[j][i];    
            for(int k=i;k<=n+1;k++) arr[j][k] -= arr[i][k] * div; //其他行减去该行
        }
    }
    ans[n] = arr[n][n+1];
		for(int i=n-1;i>=1;i--)
		{
				ans[i] = arr[i][n+1];
				for(int j=i+1;j<=n;j++)
						ans[i] -= arr[i][j] * ans[j];
		}
		for(int i=1;i<=n;i++)
				printf("%.2lf\n",ans[i]);
    return 0;
}
