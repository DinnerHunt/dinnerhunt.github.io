#include <cmath>
#include <algorithm>
using namespace std;
const double eps = 1e-8;
const int maxd = 100+5;
double arr[maxd][maxd],ans[maxd];
int n;
int main()
{
    // freopen("a.in","r",stdin);

    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=n+1;j++)
            scanf("%lf",&arr[i][j]);
    for(int i=1;i<=n;i++)
    {
        int p = i;
        for(int j=i;j<=n;j++)   //жүҫзі»ж•°жңҖеӨ§зҡ„зӮ№
            if(fabs(arr[j][i]) < fabs(arr[p][i])) p=j;

        if(fabs(arr[i][i])<=eps) 
        {
            printf("No Solution\n");
            return 0;
        }
        if(i!=p) swap(arr[i],arr[p]);
        double div = arr[i][i];
        for(int j=i+1;j<=n+1;j++) arr[i][j] /= div; //еҲ—дёӯжүҖжңүзі»ж•°е…ЁйғЁйҷӨд»ҘиҜҘж•°
        for(int j=i+1;j<=n;j++)
        {
            div = arr[j][i];

        }
    }
    ans[n] = arr[n][n+1];
