P3389 高斯消元

给定一个线性方程组，对其求解

// a 放公式，b放每条式子的值
const int N = 505;
double a[N][N],b[N],ans[N];
void Gauss(ll n)
{
    FOR(i,1,n)
    {
        ll p=i;
        FOR(k,i+1,n) if(fabs(a[k][i])>fabs(a[p][i])) p=k;
        if(i!=p) std::swap(a[i],a[p]),std::swap(b[i],b[p]);
        if(fabs(a[i][i])<=eps)
        {
            PS("No Solution\n");
            return;
        }
        FOR(k,i+1,n)
        {
            double t=a[k][i]/a[i][i]; b[k]-=t*b[i];
            FOR(j,1,n) a[k][j]-=t*a[i][j];
        }
    }
    _FOR(i,n,1)
    {
        FOR(j,i+1,n) b[i]-=ans[j]*a[i][j];
        ans[i]=b[i]/a[i][i];
    }
    FOR(i,1,n) printf("%.2lf\n",ans[i]);
}
