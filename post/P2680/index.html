「NOIP2015」运输计划

2019-05-05 #二分 #LCA

公元2044 年，人类进入了宇宙纪元。

L 国有 n 个星球，还有 n-1 条双向航道，每条航道建立在两个星球之间，这 n-1 条航道连通了 L 国的所有星球。

小 P 掌管一家物流公司， 该公司有很多个运输计划，每个运输计划形如：有一艘物流飞船需要从 uiu_iui​ 号星球沿最快的宇航路径飞行到 viv_ivi​ 号星球去。显然，飞船驶过一条航道是需要时间的，对于航道 j，任意飞船驶过它所花费的时间为 tjt_jtj​，并且任意两艘飞船之间不会产生任何干扰。

为了鼓励科技创新， L 国国王同意小 P 的物流公司参与 L 国的航道建设，即允许小P 把某一条航道改造成虫洞，飞船驶过虫洞不消耗时间。

在虫洞的建设完成前小 P 的物流公司就预接了 m 个运输计划。在虫洞建设完成后，这 m 个运输计划会同时开始，所有飞船一起出发。当这 m 个运输计划都完成时，小 P 的物流公司的阶段性工作就完成了。

如果小 P 可以自由选择将哪一条航道改造成虫洞， 试求出小 P 的物流公司完成阶段性工作所需要的最短时间是多少？

链接🔗
[NOIP 2015]运输计划

题解

用 LCA 预先处理出所有路径的距离长度,然后二分一个值最大值，找出所有超过这个值的路径，然后我们要更改的路径一定要覆盖所有超过这个值的路径，并且将路径变成 0 后路径的值一定小于二分的最大值

代码

#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 3e5+10;
struct
{
    int v,net,w;
}e[maxd<<1];
int f[maxd],eid,ans;
int n,m,a,b,c;
void insert(int x,int y,int z)
{
    e[eid].v = y;
    e[eid].net = f[x];
    e[eid].w = z;
    f[x] = eid++;
}
struct
{
    int x,y,lca,dist;
}arr[maxd];
int d[maxd],p[maxd][30],cnt[maxd],dist[maxd],num[maxd],tot;
void init()
{
    memset(f,-1,sizeof(f));
    memset(d,-1,sizeof(d));
    d[1] = 0;
    eid = 0;
}
void dfs(int x)
{
    num[++tot] = x; // dfs序
    for(int i=f[x];i+1;i=e[i].net)
        if(d[e[i].v] == -1)
        {
            d[e[i].v] = d[x] +1;
            p[e[i].v][0] = x;
            dist[e[i].v] = dist[x]+e[i].w;
            dfs(e[i].v);
        }
}
int lca(int x,int y)
{      
    if(d[x] < d[y]) swap(x,y);
    for(int i=25;i>=0;--i)
        if(d[p[x][i]] >= d[y]) x = p[x][i];
    if(x == y) return x;
    for(int i=25;i>=0;i--)
        if(p[x][i]!=p[y][i]) x = p[x][i], y = p[y][i];
    return p[x][0];

}
bool check(int mid)
{
    int ans = 0,sum=0;
    memset(cnt,0,sizeof(cnt));
    for(int i=1;i<=m;i++)
        if(arr[i].dist > mid)
        {
            cnt[arr[i].x]++,cnt[arr[i].y]++;
            cnt[arr[i].lca]-=2;
            ans = max(ans,arr[i].dist - mid);
            sum++;
        }
    if(!sum) return true;
    for(int i=n;i>=1;i--) cnt[p[num[i]][0]] += cnt[num[i]];
    for(int i=2;i<=n;i++) if(cnt[i]==sum&&dist[i] - dist[p[i][0]] >= ans) return true;
    return false;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    init();
    scanf("%d %d",&n,&m);
    for(int i=1;i<n;i++)
    {
        scanf("%d %d %d",&a,&b,&c);
        insert(a,b,c);
        insert(b,a,c);
    }
    dfs(1);
    for(int j=1;j<=25; j++)
        for(int i=1;i<=n;i++)
            p[i][j] = p[p[i][j-1]][j-1];
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d",&arr[i].x,&arr[i].y);
        arr[i].lca = lca(arr[i].x,arr[i].y);
        arr[i].dist = dist[arr[i].x] + dist[arr[i].y] - 2*dist[arr[i].lca];
    }
    int l = 0,r = 1e9,ans;
    while(l<=r)
    {
        int mid = (l+r) >> 1;
        if(check(mid))
        {
            ans = mid;
            r= mid-1;
        }
        else l = mid +1;
    }
    printf("%d",ans);
    return 0;
}
