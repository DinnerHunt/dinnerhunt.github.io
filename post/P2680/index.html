#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 3e5+10;
struct
{
    int v,net,w;
}e[maxd<<1];
int f[maxd],eid,ans;
int n,m,a,b,c;
void insert(int x,int y,int z)
{
    e[eid].v = y;
    e[eid].net = f[x];
    e[eid].w = z;
    f[x] = eid++;
}
struct
{
    int x,y,lca,dist;
}arr[maxd];
int d[maxd],p[maxd][30],cnt[maxd],dist[maxd],num[maxd],tot;
void init()
{
    memset(f,-1,sizeof(f));
    memset(d,-1,sizeof(d));
    d[1] = 0;
    eid = 0;
}
void dfs(int x)
{
    num[++tot] = x; // dfsåº
    for(int i=f[x];i+1;i=e[i].net)
        if(d[e[i].v] == -1)
        {
            d[e[i].v] = d[x] +1;
            p[e[i].v][0] = x;
            dist[e[i].v] = dist[x]+e[i].w;
            dfs(e[i].v);
        }
}
int lca(int x,int y)
{      
    if(d[x] < d[y]) swap(x,y);
    for(int i=25;i>=0;--i)
        if(d[p[x][i]] >= d[y]) x = p[x][i];
    if(x == y) return x;
    for(int i=25;i>=0;i--)
        if(p[x][i]!=p[y][i]) x = p[x][i], y = p[y][i];
    return p[x][0];

}
bool check(int mid)
{
    int ans = 0,sum=0;
    memset(cnt,0,sizeof(cnt));
    for(int i=1;i<=m;i++)
        if(arr[i].dist > mid)
        {
            cnt[arr[i].x]++,cnt[arr[i].y]++;
            cnt[arr[i].lca]-=2;
            ans = max(ans,arr[i].dist - mid);
            sum++;
        }
    if(!sum) return true;
    for(int i=n;i>=1;i--) cnt[p[num[i]][0]] += cnt[num[i]];
    for(int i=2;i<=n;i++) if(cnt[i]==sum&&dist[i] - dist[p[i][0]] >= ans) return true;
    return false;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    init();
    scanf("%d %d",&n,&m);
    for(int i=1;i<n;i++)
    {
        scanf("%d %d %d",&a,&b,&c);
        insert(a,b,c);
        insert(b,a,c);
    }
    dfs(1);
    for(int j=1;j<=25; j++)
        for(int i=1;i<=n;i++)
            p[i][j] = p[p[i][j-1]][j-1];
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d",&arr[i].x,&arr[i].y);
        arr[i].lca = lca(arr[i].x,arr[i].y);
        arr[i].dist = dist[arr[i].x] + dist[arr[i].y] - 2*dist[arr[i].lca];
    }
    int l = 0,r = 1e9,ans;
    while(l<=r)
    {
        int mid = (l+r) >> 1;
        if(check(mid))
        {
            ans = mid;
            r= mid-1;
        }
        else l = mid +1;
    }
    printf("%d",ans);
    return 0;
}
