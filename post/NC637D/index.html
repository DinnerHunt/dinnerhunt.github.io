#include <set>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
const int maxd = 1e5+10;
struct {int v,net;}e[maxd<<1];
int f[maxd],eid,n,m,col[maxd],ans[maxd];
int son[maxd],dfn[maxd],sz[maxd],cnt,top[maxd],d[maxd],fa[maxd],id[maxd];
set<int> s[maxd]; set<int>::iterator it,pre,suff;
inline int read()
{
   int f=1,x=0;char ch;
   do{ch=getchar();if(ch=='-')f=-1;}while(ch<'0'||ch>'9');
   do{x=x*10+ch-'0';ch=getchar();}while(ch>='0'&&ch<='9');
   return f*x;
}
void insert(int x,int y)
{
   e[eid].v = y;
   e[eid].net = f[x];
   f[x] = eid++;
}
void init()
{
   memset(f,-1,sizeof(f));
   eid=0;
}
void dfs(int x)
{
   sz[x] = 1;d[x] = d[fa[x]]+1;
   for(int i=f[x];i+1;i=e[i].net)
   {
       int v = e[i].v;
       if(v == fa[x]) continue;       
       fa[v] = x;
       dfs(v); sz[x] += sz[v];
       if(sz[v] > sz[son[x]]) son[x]  = v;
   }
}
void dfs1(int x,int tp)
{
   top[x] = tp; dfn[x] = ++cnt;id[cnt] = x;
   if(son[x]) dfs1(son[x],tp);
   for(int i=f[x];i+1;i=e[i].net)
   {
       int v = e[i].v; 
       
       if(v!=son[x]&&v!=fa[x])dfs1(v,v);
   }
}
int lca(int x,int y)
{
   x=id[x],y=id[y];
   while(top[x]^top[y])
   {
       if(d[top[x]]>d[top[y]])x=fa[top[x]];
       else y=fa[top[y]];
   }
   return d[x]<d[y]?x:y;
}
void add(int x)
{
   int c = col[x];ans[c]+=d[x];x = dfn[x];
   s[c].insert(x);it = s[c].find(x);
   if(it!=s[c].begin()) pre = it,pre--,ans[c] -= d[lca(*pre,*it)];
   suff = it,suff++;
   if(suff!=s[c].end())
   {
       ans[c] -= d[lca(*suff,*it)];
       if(it!=s[c].begin()) ans[c] += d[lca(*pre,*suff)];
   }
}
void del(int x)
{
   int c = col[x];ans[c]-=d[x];x = dfn[x];
   it = s[c].find(x);
   if(it!=s[c].begin()) pre = it,pre--,ans[c] += d[lca(*pre,*it)];
   suff = it,suff++;
   if(suff!=s[c].end())
   {
       ans[c] += d[lca(*suff,*it)];
       if(it!=s[c].begin()) ans[c] -= d[lca(*pre,*suff)];
   }
   s[c].erase(x);
}
int main()
{
   // freopen("a.in","r",stdin);
   // freopen("k.out","w",stdout);
   init();
   scanf("%d %d",&n,&m);
   for(int i=1;i<=n-1;i++)
   {
       int a,b;a = read(),b = read();
       insert(a,b); insert(b,a);
   }
   dfs(1);dfs1(1,1);
   //for(int i=1;i<=n;i++) printf("%d ",fa[i]);
   for(int i=1;i<=n;i++){col[i] =read();add(i);}
   for(int i=1;i<=m;i++)
   {
       int a,x;a = read(),x = read();
       if(a==1)
       {
           del(x);
           col[x] = read();add(x);
       }
       else
       {
           if(!s[x].size()) printf("0\n");
           else printf("%d\n",ans[x] - d[lca(*s[x].begin(),*s[x].rbegin())]+1);
       }
   }
   return 0;
}
