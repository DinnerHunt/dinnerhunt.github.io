最大流算法 Dinic

#include <cstdio>
#include <algorithm>
#include <cstring>
#include <queue>
using namespace std;
const int maxd = 505;   //点数
const int maxn = 1e5+10; //边数
const int inf = 1e9+10;
struct
{
    int v,net,c;
}e[maxn];
int f[maxd],eid;
void init()
{
    memset(f,-1,sizeof(f));
    eid = 0;
}
void insert(int x,int y,int c)
{
    e[eid].v = y;
    e[eid].net = f[x];
    e[eid].c = c;
    f[x] = eid++;
}
void addedge(int x,int y,int z)
{
    insert(x,y,z);
    insert(y,x,0);
}
int s,t;
int d[maxd],in[maxd],out[maxd],k[maxd];
bool bfs()
{
    memset(d,-1,sizeof(d));
    queue<int> que;
    que.push(s);
    d[s] = 0;
    while(!que.empty())
    {
        int x = que.front();
        que.pop();
        for(int i=f[x];i!=-1;i=e[i].net)
        {
            int v = e[i].v;
            if(e[i].c > 0 && d[v] ==-1)
            {
                que.push(v);
                d[v] = d[x] +1;
            }
        }
    }
    return (d[t] != -1);
}
int dfs(int x,int flow)
{
    if( x== t) return flow;
    int res = 0;
    for(int i = f[x];i+1;i=e[i].net)
    {
        int v = e[i].v;
        if(e[i].c >0 && d[x]+1 == d[v])
        {
            int tmp = dfs(v,min(flow,e[i].c));
            flow -= tmp;
            e[i].c -= tmp;
            res += tmp;
            e[i^1].c += tmp;
            if(!flow) break;
        } 
    }
    if(res == 0) d[x] = -1;
    return res;
}
int maxflow()
{
    int res = 0;
    while(bfs()) res += dfs(s,inf);
    return res;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    init();
    return 0;
}
