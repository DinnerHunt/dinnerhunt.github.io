#include <cstdio>
#include <algorithm>
#include <cstring>
const int maxd = 3000+10;
struct edge {
    int v, net;
} e[maxd<<2];
int p[maxd], eid;
void init(){
  memset(p,-1,sizeof(p));
  eid=0;
}
void insert(int a,int b){
    e[eid].v = b;
    e[eid].net = p[a];
    p[a] = eid++;
}
char ans[maxd];
int vis[maxd],n,m;
void dfs(int x)
{ 
    vis[x] = 1;
    for(int i=p[x];i+1;i =e[i].net) 
        if(!vis[e[i].v]) dfs(e[i].v);
}
bool check(int x)
{
    memset(vis,0,sizeof(vis));
    dfs(x);
    for(int i=1;i<=n;i++)
        if(vis[i*2] && vis[i*2+1]) return false;
    return true;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    init();
    scanf("%d %d",&n,&m);
    for(int i=1;i<=m;i++)
    {
        int x,y; char a,b;
        scanf("%d %c %d %c",&x,&a,&y,&b);
        x = 2 * x + (a == 'Y');
        y = 2 * y + (b == 'Y');
        insert(x^1,y);
        insert(y^1,x);
    }
    for(int i=1;i<=n;i++)
    {
        bool y = check(i*2+1), n = check(i*2);
        if(!y && !n)
        {
            printf("IMPOSSIBLE");
            return 0;
        }
        if(n&&y) ans[i] = '?';
        if(n && !y) ans[i] = 'N';
        if(!n&&y) ans[i] = 'Y';
    }
    printf("%s",ans+1);
    return 0;
}
