「Template」Tarjan

2019-05-04

缩点

V G[N];
int dfn[N], low[N],idx = 0;
int belong[N], scc = 0;
V kuai[N];
int s[N], top = 0;
bool in_stack[N];
void tarjan(int u)
{
    dfn[u] = low[u] = ++idx;
    s[top++] = u;
    in_stack[u] = true;
    for(auto v: G[u])
    {
        if (!dfn[v])
        {
            tarjan(v);
            low[u] = min(low[u], low[v]);
        }
        else if (in_stack[v])
            low[u] = min(low[u], dfn[v]);
    }
    if(dfn[u] == low[u])
    {
        ++scc;
        do
        {
            belong[s[--top]] = scc;
            kuai[scc].pb(s[top]);
            in_stack[s[top]] = false;
        } while (s[top] != u);
    }
}

割点

V G[N];
int dfn[N],low[N],tot= 0;
bool iscut[N];
void tarjan(int x,int fa)
{
    int child = 0;
    dfn[x] = low[x] = ++tot;
    for(auto v:G[x])
    {
        if(!dfn[v])
        {
            tarjan(v,fa);
            low[x] = min(low[v],low[x]);
            if(dfn[x] <= low[v] && x!=fa) iscut[x] = 1;
            if(x==fa) child++;
        }
        low[x] = min(low[x],dfn[v]);
    }
    if(x == fa && child>=2) iscut[x] = true;
}

边双

VP G[N];
ll low[N],dfn[N],tot,top,stk[N];
ll belong[N],scc;
void tarjan(ll x,ll fa)
{
    low[x] = dfn[x] = ++tot;
    stk[top++] = x;
    for(auto p: G[x])
    {
        ll v = p.ST, id = p.ND;
        if(dfn[v])
        {
            if(dfn[v] < dfn[x] && fa != id) low[x]=min(low[x],dfn[v]);
        }
        else
        {
            tarjan(v,id);
            low[x] = min(low[x],low[v]);
        }
    }
    if(dfn[x] == low[x])
    {
        ++scc;
        do {belong[stk[--top]] = scc;} while (stk[top] != x);
    }
}
