#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 5e4+10;
int m,n,ans,k,t;
int mu[maxd], prime[maxd],summu[maxd],cnt;
long long f[maxd];
bool vis[maxd];
void init() 
{
    mu[1] = 1;
    for(int i=2;i<maxd;i++)
    {
        if(!vis[i]) prime[++cnt] = i, mu[i] = -1;
        for(int j=1;j<=cnt && i*prime[j] <= maxd;j++)
        {
            vis[i*prime[j]] = 1;
            if(i%prime[j] == 0) break;
            mu[i*prime[j]] = -mu[i]; 
        }
    }
    for(int i=1;i<maxd;i++) summu[i] = summu[i-1] + mu[i];
    for(int i=1;i<=5e4;i++)   //йў„еӨ„зҗҶh(i) 
        for(int l=1,r;l<=i;l = r+1) r = i/(i/l),f[i]+=1LL * (r-l+1) * (i/l);
}
long long sum(int m,int n)  //ж•°и®әеҲҶеқ—
{
    long long t = min(n, m), ans = 0;
    for (int i = 1, j = 1; i <= t; i = j + 1) 
    {
        j = min(n / ( n / i), m / (m / i));
        ans += (summu[j] - summu[i - 1]) * f[n / i] * f[m / j];
    }
    return ans;
}
int main()
{
    // freopen("a.in","r",stdin);

    init();scanf("%d",&t);
    for(int i=1;i<=t;i++)
    {
        scanf("%d %d",&n,&m);
        printf("%lld\n",sum(n,m));
    }
    return 0;
}
