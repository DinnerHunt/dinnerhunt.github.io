「HDU 5988」Coding Contest

2019-11-29 #网络流 #最小费用最大流

A coding contest will be held in this university, in a huge playground. The whole playground would be divided into N blocks, and there would be M directed paths linking these blocks. The i-th path goes from the ui-th block to the vi-th block. Your task is to solve the lunch issue. According to the arrangement, there are si competitors in the i-th block. Limited to the size of table, bi bags of lunch including breads, sausages and milk would be put in the i-th block. As a result, some competitors need to move to another block to access lunch. However, the playground is temporary, as a result there would be so many wires on the path.

For the i-th path, the wires have been stabilized at first and the first competitor who walker through it would not break the wires. Since then, however, when a person go through the i - th path, there is a chance of pi to touch
the wires and affect the whole networks. Moreover, to protect these wires, no more than ci competitors are allowed to walk through the i-th path.
Now you need to find a way for all competitors to get their lunch, and minimize the possibility of network crashing.

题解

按最小费用最大流建边
然后因为概率是乘法，但是费用流部分统计答案用的是加法，所以先取个对数，x∗y=exp(lg(x)+ln(y))x * y = exp(lg(x) + ln(y))x∗y=exp(lg(x)+ln(y)) 等于对数的数加，然后再跑费用流

代码

#include<cstdio>
#include<algorithm>
#include<cstring>
#include<queue>
#include<vector>
#include <cmath>
using namespace std;
typedef long long ll;
const int inf = 0x3f3f3f3f;
const int maxd = 205;
const double eps = 0.000001;
struct Edge
{
    int to, rev;
    int flow;
    double cost;
};

int num; // 默认点的范围从0开始，num为最大点数
vector<Edge> edge[maxd];
double h[maxd],dis[maxd];
int pre_node[maxd], pre_edge[maxd];

void add(int u, int v, int flow, double cost)
{
    edge[u].push_back((Edge){ v,(int)edge[v].size(),flow,cost });
    edge[v].push_back((Edge){ u,(int)edge[u].size() - 1,0,-cost });

}
typedef pair<double,int> P;
void min_cost_flow(int s, int t, double& min_cost, int& max_flow)
{
    fill(h, h + num+1, 0.0);
    min_cost = 0.0 ; max_flow = 0.0;
    int tot = inf;
    while (tot > 0)
    {
        priority_queue<P, vector<P>, greater<P> > q;
        fill(dis, dis + num +1, inf);
        dis[s] = 0.0; q.push(P( dis[s],s));
        while (!q.empty())
        {
            int u = q.top().second;
            int dist = q.top().first;
            q.pop();
            if (dis[u] < dist)continue;
            for (int i = 0;i < edge[u].size(); i++)
            {
                Edge e = edge[u][i];
                if (e.flow > 0 && dis[e.to] - (dis[u] + e.cost + h[u] - h[e.to]) >= eps )
                {

                    dis[e.to] = dis[u] +e.cost + h[u] - h[e.to];
                    pre_node[e.to] = u;
                    pre_edge[e.to] = i;
                    q.push(P(dis[e.to],e.to ));

                }
            }
        }

        if (fabs(dis[t] - inf) <= eps) return;
        //  这里注意点的范围
        for (int i = 0;i <= num;i++) {
            h[i] += dis[i];
            //printf("%lf ",dis[i]);
        }
        //putchar(10);
        int flow = tot;
        for (int i = t; i != s; i = pre_node[i])
            flow = min(flow, edge[pre_node[i]][pre_edge[i]].flow);

        for (int i = t; i != s; i = pre_node[i])
        {
            Edge& e = edge[pre_node[i]][pre_edge[i]];
            e.flow -= flow;
            edge[i][e.rev].flow += flow;
        }
        tot -= flow;
        max_flow += flow;
        min_cost += flow * h[t];

        //printf("%d %lf %d %lf %d %lf\n",tot,dis[t],flow,min_cost,t,h[t]);
    }
}
void init()
{
    for(int i=0;i<=num;i++) edge[i].clear();
}
int main()
{
//    freopen("a.in","r",stdin);
//    freopen("k.out","w",stdout);
    int s,t,T;
    scanf("%d",&T);
    while(T--)
    {
        int n,m;
        scanf("%d %d",&n,&m);init();
        s = 0, t = n+1; num =  t;
        for(int i=1;i<=n;i++)
        {
            int a,b;scanf("%d %d",&a,&b);
            int x=min(a,b);
            a-=x,b-=x;
            if(a>0) add(s,i,a,0);
            else if(b>0) add(i,t,b,0);
            //printf("%d %d %d\n",a,b,a-b);
        }
        for(int i=1;i<=m;i++)
        {
            int a,b,c; double d;
            scanf("%d %d %d %lf",&a,&b,&c,&d);
            if(c) add(a,b,1,0);
            if(c>1)add(a,b,c-1,-1*log(1.0-d));
        }
        double min_cost = 0.0;
        int max_flow= 0;
        min_cost_flow(s,t,min_cost,max_flow);
        printf("%.2lf\n",1.0 - exp(-1.0 * min_cost));
    }
    return 0;
}
