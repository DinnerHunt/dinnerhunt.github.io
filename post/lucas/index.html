组合数的求法

$O(n^2)$ 递推

Cnm=Cn−1m+Cn−1m−1C_n^m = C_{n-1}^{m} + C_{n-1}^{m-1} C​n​m​​=C​n−1​m​​+C​n−1​m−1​​

O(n)O(n)O(n) 线性逆元法

fac[i]=i!inv[i]=inv[i]!Cnm=fac[n]⋅inv[n−m]⋅inv[m] fac[i] = i! inv[i] = inv[i]! C_n^m = fac[n] \cdot inv[n-m] \cdot inv[m] fac[i]=i!inv[i]=inv[i]!C​n​m​​=fac[n]⋅inv[n−m]⋅inv[m]

O(n)O(n)O(n) 卢卡斯定理 当 ppp 为质数时

Cnm=Cn/pm/p⋅Cn%pm%pC_n^m = C_{n/p}^{m/p} \cdot C_{n \% p}^{m \% p} C​n​m​​=C​n/p​m/p​​⋅C​n%p​m%p​​

#include <cstdio>
using namespace std;
const int maxd = 1e4+10;
long long t,n,m,p,fac[maxd];
long long pow(long long x,long long y)
{
    long long now = 1;
    while(y)
    {
        if(y&1) (now *= x)%=p;
        (x *= x)%=p;
        y>>=1;
    }
    return now;
}
long long C(long long n,long long m)
{
    if(n<m) return 0;
    long long a = 1, b =1;
    for(long long i = n-m+1; i<=n; i++) (a*=i)%=p;
    for(long long i = 1; i<=m; i++) (b*=i)%=p;
    return a * pow(b,p-2) %p;
}
long long lucas(long long n,long long m)
{
    if(!m) return 1;
    return C(n%p,m%p) * lucas(n/p,m/p)%p;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%lld",&t); fac[1] = 1;
    for(int i=1;i<=t;i++)
    {
        scanf("%lld %lld %lld",&n,&m,&p);
        printf("%lld\n",lucas(n,m));   
    }
    return 0;
}
