#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
const int MAX_N = 2005;
const int MAX_M = 2005*2005;
struct edge {
    int v, next;
} e[MAX_M];
int p[MAX_N], eid;
void init(){
  memset(p,-1,sizeof(p));
  eid=0;
}
void insert(int a,int b){
    e[eid].v = b;
    e[eid].next = p[a];
    p[a] = eid++;
}
int dfn[MAX_N], low[MAX_N];  
int idx = 0;
int belong[MAX_N], scc = 0; 
int s[MAX_N], top = 0,x,n; 
bool in_stack[MAX_N];
void tarjan(int u) {
        dfn[u] = low[u] = ++idx;
        s[top++] = u;
        in_stack[u] = true;
        for (int i = p[u]; i != -1; i = e[i].next) 
        {
            int v = e[i].v;
            if (!dfn[v]) {
                tarjan(v);
                low[u] = min(low[u], low[v]);
            } else if (in_stack[v]) {
                low[u] = min(low[u], dfn[v]);
            }
        }
        if (dfn[u] == low[u]) {
            ++scc;
            do {
                belong[s[--top]] = scc;
                in_stack[s[top]] = false;
            } while (s[top] != u);
        }
}
int main()
{

        {
            scanf("%d",&x);
            if(x) insert(i,j);
        }
    for(int i=1;i<=n;i++) if(!dfn[i]) tarjan(i);
    if(scc==1) printf("YES");
    else printf("NO");
    return 0;
}
