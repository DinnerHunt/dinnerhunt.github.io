int CRT(int a[], int m[], int n) {
    int M = 1;
    int ans = 0;
    for(int i = 1; i <= n; i++) {
        M *= m[i];
    }
    for(int i = 1; i <= n; i++) {
        int x, y;
        int Mi = M / m[i];
        exgcd(Mi, m[i], x, y);
        ans = (ans + Mi * x * a[i]) % M;
    }
    if(ans < 0) ans += M;
    return ans;
}

long long crt(long long n) {
    long long M = b[1], ans = a[1], x, y;
    for (int i = 2; i <= n; i++) {
        long long z = (a[i] - ans%b[i]+b[i])%b[i];
        long long k = exgcd(M, b[i], x, y);
        if (z % k)
            return -1;
        long long c = b[i] / k;   
        x = mul(x,z/k,c);
        ans += x * M;  
        M = M * c;
        ans = (ans%M+M)%M;
    }
    return (ans % M + M) % M;
}
