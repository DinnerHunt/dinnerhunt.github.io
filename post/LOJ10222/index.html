求 T(x)%mT(x) \% mT(x)%m 的值

题解

已知

T(n)=F(1)+2⋅F(2)+3⋅F(3)+⋯+n⋅F(n)T(n) = F(1) + 2\cdot F(2) + 3\cdot F(3) + \cdots + n\cdot F(n)T(n)=F(1)+2⋅F(2)+3⋅F(3)+⋯+n⋅F(n)
n⋅S(n)=n⋅F(1)+n⋅F(2)+n⋅F(3)+⋯+n⋅F(n)n \cdot S(n) = n \cdot F(1) + n\cdot F(2) + n\cdot F(3) + \cdots + n\cdot F(n)n⋅S(n)=n⋅F(1)+n⋅F(2)+n⋅F(3)+⋯+n⋅F(n)


设
P(n)=n⋅S(n)−T(n)=(n−1)⋅F(1)+(n−2)⋅F(2)+⋯+1⋅F(n);P(n) = n \cdot S(n) - T(n) = (n-1) \cdot F(1) + (n-2)\cdot F(2) + \cdots +1 \cdot F(n);P(n)=n⋅S(n)−T(n)=(n−1)⋅F(1)+(n−2)⋅F(2)+⋯+1⋅F(n);
$T(n) =n \cdot S(n) - P(n) $


然后我们看 P(n)P(n)P(n) ,这里用到了 S(n)S(n)S(n)

P(n+1)=(n)⋅F(1)+(n−1)⋅F(2)+⋯+2⋅F(n);P(n+1) = (n) \cdot F(1) + (n-1)\cdot F(2) + \cdots + 2 \cdot F(n);P(n+1)=(n)⋅F(1)+(n−1)⋅F(2)+⋯+2⋅F(n);
P(n+1)=P(n)+S(n)P(n+1) = P(n) + S(n)P(n+1)=P(n)+S(n)


再看 S(n)S(n)S(n) ,这里用到了 F(n+1)F(n+1)F(n+1)

S(n+1)=S(n)+F(n+1)S(n+1) = S(n) + F(n+1)S(n+1)=S(n)+F(n+1)


然后 F(n)F(n)F(n) 这里我们可以巧妙的构造一个矩阵求解

F(n+2)=F(n+1)+F(n)F(n+2) = F(n+1) + F(n)F(n+2)=F(n+1)+F(n)

F(n+1)=F(n+1)+0F(n+1) = F(n+1) + 0F(n+1)=F(n+1)+0


最后我们把上面的式子总结下

P(n+1)=1⋅P(n)+1⋅S(n)+0⋅F(n+1)+0⋅F(n)P(n+1) = 1 \cdot P(n) + 1 \cdot S(n) + 0 \cdot F(n+1) + 0 \cdot F(n)P(n+1)=1⋅P(n)+1⋅S(n)+0⋅F(n+1)+0⋅F(n)

S(n+1)=0⋅P(n)+1⋅S(n)+1⋅F(n+1)+0⋅F(n)S(n+1) = 0 \cdot P(n) + 1 \cdot S(n) + 1 \cdot F(n+1) + 0 \cdot F(n)S(n+1)=0⋅P(n)+1⋅S(n)+1⋅F(n+1)+0⋅F(n)

F(n+2)=0⋅P(n)+0⋅S(n)+1⋅F(n+1)+1⋅F(n)F(n+2) = 0 \cdot P(n) + 0 \cdot S(n) + 1 \cdot F(n+1) + 1 \cdot F(n)F(n+2)=0⋅P(n)+0⋅S(n)+1⋅F(n+1)+1⋅F(n)

F(n+1)=0⋅P(n)+0⋅S(n)+1⋅F(n+1)+0⋅F(n)F(n+1) = 0 \cdot P(n) + 0 \cdot S(n) + 1 \cdot F(n+1) + 0 \cdot F(n)F(n+1)=0⋅P(n)+0⋅S(n)+1⋅F(n+1)+0⋅F(n)

然后我们巧妙设计一个矩阵来表示他们两项之间的变化

然后矩阵快速幂加速求解即可

#include <cstdio
#include <algorithm
#define ll long long
using namespace std;
long long ans;
struct matrix
{
    int n,m;
    ll a[105][105];
};
matrix matrix_mul(matrix A,matrix B,ll mod)
{
    matrix C;
    C.n = A.n;
    C.m = B.m;
    for(int i=1;i<=A.n;i++)
        for(int j=1;j<=B.m;j++)
        {
            C.a[i][j] = 0;
            for(int k = 1;k<=A.m;++k)
                (C.a[i][j] += (A.a[i][k] * B.a[k][j])%mod)%=mod;
        }
    return C;
}
matrix unit(int n)
{
    matrix res;
    res.n = res.m = n;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)
            res.a[i][j] = (i==j);
    return res;
}
matrix init()
{
    matrix res;
    for(int i=1;i<=4;i++)
        for(int j=1;j<=4;j++)
            res.a[i][j] = 0;
    res.n = res.m = 4;
    res.a[1][1] = res.a[1][2] =
    res.a[2][2] = res.a[2][3] =
    res.a[3][3] = res.a[3][4] = res.a[4][3] = 1;
    return res;
}
matrix matrix_pow(matrix A,ll n,ll mod)
{
    matrix res = unit(4), temp = A;
    for(;n;n=1)
    {
        if(n&1) res = matrix_mul(res,temp,mod);
        temp = matrix_mul(temp,temp,mod);
    }
    return res;
}

int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    matrix A = init();
    ll n,m;
    scanf("%lld %lld",&n,&m);
    if(n<=2) printf("%lld",n==1?1:3%n);
    else
    {
        A = matrix_pow(A,n-1,m);
        // P[1] = 0, S[1] = F[2] = F[1] =1
        ll P = ( A.a[1][2] + A.a[1][3] + A.a[1][4])%m;
        ll S = ( A.a[2][2] + A.a[2][3] + A.a[2][4])%m;
        printf("%lld",(S*n-P + m)%m);
    }

    return 0;
}
