#include <cstdio>
#include <cstring>
#include <queue>
#include <algorithm>
using namespace std;
const int maxd = 1e6+10;
char str[10000010],arr[maxd];
int slen[maxd],cnt[maxd];
int id(char ch)
{
    switch(ch)
    {
        case 'E' : return 0;
        case 'S' : return 1;
        case 'W' : return 2;
        case 'N' : return 3;
    }
    return 0;
}
struct 
{
    int ch[maxd][4],fail[maxd],tot,last[maxd];
    bool vis[maxd];
    void init()
    {
        memset(ch,0,sizeof(ch));
        memset(fail,0,sizeof(fail));
        tot = 0;
    }
    void insert(char *str,int k)
    {
        int p = 0;
        for(int i=0;str[i];i++)
        {
            int k = id(str[i]);
            if(!ch[p][k])
            {
                ch[p][k] = ++tot;
                last[tot] = p;
            }
            p = ch[p][k];
        }
        cnt[k] = p;
    }
    void build()
    {
        memset(fail,0,sizeof(fail));
        queue<int> que;
        for(int i=0;i<4;i++) if(ch[0][i]) que.push(ch[0][i]);
        while(!que.empty())
        {
            int p = que.front(); que.pop();
            for(int i=0;i<4;i++)
            { 
                if(ch[p][i])
                {
                    fail[ch[p][i]] = ch[fail[p]][i];
                    que.push(ch[p][i]);
                }
                else ch[p][i] = ch[fail[p]][i]; 
            }
        }
    }
    int solve(char *str)
    {
        int p = 0,ans = 0;
        for(int i=0;str[i];i++)
        {
            p = ch[p][id(str[i])];
            int k = p;
            while(k)
            {
                vis[k] = 1;
                k = fail[k];
            }
        }
        return ans;
    }
    int find(int x)
    {
        int p = cnt[x];
        for(int i=slen[x];i>0;i--)
        {
            if(vis[p]) return i;
            p = last[p];
        }
        return 0;
    }
}AC;
int main()
{   
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    int n,m;
    scanf("%d %d",&n,&m);
    scanf("%s",str);
    for(int i=1;i<=m;i++)
    {
        scanf("%s",arr);
        slen[i] = strlen(arr);
        AC.insert(arr,i);
    }
    AC.build();
    AC.solve(str);
    for(int i=1;i<=m;i++)
        printf("%d\n",AC.find(i));
    return 0;
}
