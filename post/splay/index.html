普通平衡树

#include <cstdio>
#include <algorithm>
using namespace std;
const int maxd = 200005;
int ch[maxd][2],val[maxd],cnt[maxd],par[maxd],size[maxd],rev[maxd];
int root,ncnt;

//判断自身在父节点的左右
int chk(int x){         
    return ch[par[x]][1] == x;
}
//更新数组
void pushup(int x){
    size[x] = size[ch[x][0]] + size[ch[x][1]] + cnt[x];
}
//旋转操作
void rotate(int x) {    //将x节点往上提一层
    int y = par[x], z = par[y], k = chk(x), w = ch[x][k^1]; 
    // 如果在父节点左边，则取右儿子，不然则取左儿子，为了保持中序顺序不变
    ch[y][k] = w; par[w] = y;   
    ch[z][chk(y)] = x; par[x] = z;  
    ch[x][k^1] = y; par[y] = x; //将原本属于儿子节点给y
    pushup(y); pushup(x);   //更新节点数量
}
//伸展操作
void splay(int x, int goal = 0) {   //将x节点提升到goal节点下
    while (par[x] != goal) {    
        int y = par[x], z = par[y];
        if (z != goal) {        //  如果爷爷节点不是指定节点，那么得选择2次
            if (chk(x) == chk(y)) rotate(y);    //如果是一条直线，那么要先旋转父亲节点
            else rotate(x); 
        }
        rotate(x);   //再选择一次       
    }
    if (!goal) root = x;
}

void find(int x){   //将x的前驱提到根
    if(!root) return;   //  如果根为0，则退出
    int cur = root;     
    while(ch[cur][x>val[cur]] && x!=val[cur])
        cur = ch[cur][x>val[cur]];
    splay(cur);     //将cur提到根
}  

void insert(int x){
    int cur = root, p=0;
    while(cur && val[cur]!=x){
        p = cur;
        cur = ch[cur][x>val[cur]];
    }
    if(cur) cnt[cur]++;
    else {
        cur = ++ncnt;
        if(p) ch[p][x>val[p]] = cur;
        ch[cur][0] = ch[cur][1] = 0;
        val[cur] = x;par[cur] = p;
        cnt[cur] = size[cur] = 1;
    }
    splay(cur);
}

int kth(int k){
    int cur = root;
    while(1){
        //pushdown(cur);
        if(ch[cur][0] && k <= size[ch[cur][0]])  //往左
            cur = ch[cur][0];
        else if(k > size[ch[cur][0]] + cnt[cur]){ //往右
            k -= size[ch[cur][0]] + cnt[cur];
            cur = ch[cur][1];
        }
        else return cur;
    }
}
int pre(int x){ //提到根后，左儿子的最右儿子
    find(x);
    if(val[root] < x) return root;
    int cur = ch[root][0];
    while(ch[cur][1]) cur = ch[cur][1];
    return cur;
}
int succ(int x){ //提到根后，右儿子的最左儿子
    find(x);
    if(val[root] > x) return root;
    int cur = ch[root][1];
    while(ch[cur][0]) cur = ch[cur][0];
    return cur;
}
void remove(int x){ 
    int last = pre(x),next = succ(x);
    splay(last);splay(next,last);   // 将前驱挪到根，将后驱挪到他的儿子，一定会在右儿子
    int del = ch[next][0];          // 这时后驱一定只有一个左儿子，因为一个数的前驱和后驱之间只有它自己
    if(cnt[del] > 1){
        cnt[del]--;
        splay(del);
    }
    else ch[next][0] = 0;
}
inline int read(){
    int f=1,x=0;char ch;
    do{ch=getchar();if(ch=='-')f=-1;}while(ch<'0'||ch>'9');
    do{x=x*10+ch-'0';ch=getchar();}while(ch>='0'&&ch<='9');
    return f*x;
}
inline int print(int x){
    if(x<0){putchar('-');x=-x;}  
    if(x>=10){print(x/10);}
    putchar(x%10+'0');
}
int main(){
    int n = read();
    insert(0x3f3f3f3f);
    insert(0xcfcfcfcf);
    for(int i=1;i<=n;i++){
        int x = read(),y=read();
        switch(x){
            case 1:
                insert(y);
                break;
            case 2:
                remove(y);
                break;
            case 3:
                find(y);
                print(size[ch[root][0]]);
                putchar(10);
                break;
            case 4:
                print(val[kth(y+1)]);
                putchar(10);
                break;
            case 5:
                print(val[pre(y)]);
                putchar(10);
                break;
            case 6:
                print(val[succ(y)]);
                putchar(10);
                break;
        }
    }
    return 0;
}

#include <algorithm>
using namespace std;
const int maxd = 200005;
int ch[maxd][2],val[maxd],cnt[maxd],par[maxd],size[maxd],rev[maxd];
int root,ncnt,n,m,flag;

void find(int x){   //将x的前驱提到根
    if(!root) return;   //  如果根为0，则退出
    int cur = root;     
    while(ch[cur][x>val[cur]] && x!=val[cur])

    splay(cur);

void pushdown(int x){
    if(rev[x]){
        swap(ch[x][0],ch[x][1]);
        rev[ch[x][0]] ^=1;
        rev[ch[x][1]] ^=1;
        rev[x] = 0;
    }
}
int kth(int k){
    int cur = root;
    while(1){
        pushdown(cur);
        if(ch[cur][0] && k <= size[ch[cur][0]])  //往左
            cur = ch[cur][0];
        else if(k > size[ch[cur][0]] + cnt[cur]){ //往右
            k -= size[ch[cur][0]] + cnt[cur];
            cur = ch[cur][1];
        }
        else return cur;
    }
}
int pre(int x){ //提到根后，左儿子的最右儿子
    find(x);
    if(val[root] < x) return root;
    int cur = ch[root][0];
    while(ch[cur][1]) cur = ch[cur][1];
    return cur;
}
int succ(int x){ //提到根后，右儿子的最左儿子
    find(x);
    if(val[root] > x) return root;
    int cur = ch[root][1];
    while(ch[cur][0]) cur = ch[cur][0];
    return cur;
}
void remove(int x){ 
    int last = pre(x),next = succ(x);
    splay(last);splay(next,last);   // 将前驱挪到根，将后驱挪到他的儿子，一定会在右儿子
    int del = ch[next][0];          // 这时后驱一定只有一个左儿子，因为一个数的前驱和后驱之间只有它自己
    if(cnt[del] > 1){
        cnt[del]--;
        splay(del);
    }
    else ch[next][0] = 0;
}
void reverse(int l,int r){
    int x = kth(l),y =kth(r+2);
    splay(x);splay(y,x);
    rev[ch[y][0]] ^= 1;
}
void output(int x) {
    pushdown(x);
    if (ch[x][0]) output(ch[x][0]);
    if (val[x] && val[x] <= n) printf("%d ", val[x]);
    if (ch[x][1]) output(ch[x][1]);
}

    n = read(),m=read();
    for(int i=0;i<=n+1;i++) insert(i);
    for(int i=1;i<=m;i++){
        int a = read(),b=read();

    }
