const ll maxd = 2e6+10;

V G[maxd];
ll dfn[maxd],low[maxd],belong[maxd],s[maxd],top,scc,idx;
ll m0,n,m,m1;
bool ins[maxd];

void tarjan(int u)
{
    dfn[u] = low[u] = ++idx;
    s[top++] = u;
    ins[u] = true;
    for(auto v:G[u])
    {
        if (!dfn[v])
        {
            tarjan(v);
            low[u] = min(low[u], low[v]);
        }
        else if (ins[v])
            low[u] = min(low[u], dfn[v]);
    }
    if(dfn[u] == low[u])
    {
        ++scc;
        do
        {
            belong[s[--top]] = scc;
            ins[s[top]] = false;
        } while (s[top] != u);
    }
}
