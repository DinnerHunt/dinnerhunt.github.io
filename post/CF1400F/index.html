const ll N = 1e5+10;
ll a[N],n,x;
string s;
int ch[N][10],fail[N],sz[N],tot;
void ins(string s)
{
//    cout << s << endl;
    ll p = 0;
    for(auto w:s)
    {
        ll v = w - '0';
        if(!ch[p][v]) ch[p][v] = ++tot;
        p = ch[p][v];
    }
    sz[p]++;
}
void build()
{
    queue<ll> q; ll p =0;
    FOR(i,1,9)
        if(ch[0][i]) q.push(ch[0][i]),fail[ch[0][i]] = 0;
    while(!q.empty())
    {
        p = q.front(); q.pop();
        FOR(i,1,9)
        {
            if(ch[p][i]) fail[ch[p][i]] = ch[fail[p]][i],q.push(ch[p][i]);
            else ch[p][i] = ch[fail[p]][i];
        }
    }
}
