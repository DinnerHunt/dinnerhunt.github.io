#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
long long t,n,m,p;
const long long inf = 1e18;
long long phi(long long x)
{
    long long ans = x;
    for(int i=2;i*i<=x;i++)
    {
        if(x%i==0)
        {
            ans = ans/i *(i-1);
            while(x%i == 0) x/=i;
        }
    }
    if(x>1) ans = ans/x * (x-1);
    return ans;
}
long long pow(long long a,long long b, long long mod)
{
    long long ans = 1ll;
    while(b)
    {
        if(b&1) ans = mul(ans,a,mod);
        a = a*a%mod;
        b>>=1;
    }
    return ans;
}
long long find(long long x)
{
    long long ans = x;
    for(long long i = 1;i*i<=x;i++)
    {
        if(x%i==0)
        {
            if(pow(10,i,9*p) == 1) ans = min(ans,i);
            if(pow(10,x/i,9*p) == 1) ans = min(ans,x/i);
        }
    }
    return ans;
}
long long pri[50],c[50],tot;
void divide(long long n)
{
    tot = 0;
    for(int i=2;i*i<=n;i++)
    {
        if(n%i == 0)
        {
            pri[++tot] = i,c[tot]=0;
            while(n%i==0) c[tot]++, n/=i;
        }
    }
    if(n>1) pri[++tot] = n ,c[tot]=1;
}
int main()
{
    // freopen("a.in","r",stdin);

    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld %lld",&p,&n,&m);
        if(p%2==0||p%5==0)
        {
            printf("0\n");
            continue;
        }
        long long d = find(phi(9*p));
        //printf("%lld \n",phi(9*p));
        divide(d);
        long long ans = 0ll;
        for(int j=1;j<=min(m,30ll);j++)
        {
            long long g =1ll;
            for(int i=1;i<=tot;i++)
                g *=  pow(pri[i],((c[i]-1)/j+1),inf);
            ans += n/g;
        }
        long long g = 1ll;
        for(int i=1;i<=tot;i++) g*=pri[i];
        if(m>30)
        ans += (long long) (m-30) * (n/g);
        printf("%lld\n",ans);
    }
    return 0;
}
