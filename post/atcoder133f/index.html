void dfs(int x,int p = 0)
{
    fa[x] = p;d[x] = d[p]+1;

        w[v] = e[i].c;
        col[v] = e[i].col;
        len[v] = 1ll * len[x] + w[v];
        dfs(v,x);

        sz[x] += sz[v];
    }
}
void dfs1(int x,int p = 1)

int lca(int x,int y)

}
