Let's define рқ‘қрқ‘–(рқ‘ӣ) as the following permutation: [рқ‘–,1,2,вҖҰ,рқ‘–вҲ’1,рқ‘–+1,вҖҰ,рқ‘ӣ]. This means that the рқ‘–-th permutation is almost identity (i.e. which maps every element to itself) permutation but the element рқ‘– is on the first position. Examples:

рқ‘қ1(4)=[1,2,3,4];
рқ‘қ2(4)=[2,1,3,4];
рқ‘қ3(4)=[3,1,2,4];
рқ‘қ4(4)=[4,1,2,3].
You are given an array рқ‘Ҙ1,рқ‘Ҙ2,вҖҰ,рқ‘Ҙрқ‘ҡ (1вүӨрқ‘Ҙрқ‘–вүӨрқ‘ӣ).

Let рқ‘қрқ‘ңрқ‘ (рқ‘қ,рқ‘Јрқ‘Һрқ‘ҷ) be the position of the element рқ‘Јрқ‘Һрқ‘ҷ in рқ‘қ. So, рқ‘қрқ‘ңрқ‘ (рқ‘қ1(4),3)=3,рқ‘қрқ‘ңрқ‘ (рқ‘қ2(4),2)=1,рқ‘қрқ‘ңрқ‘ (рқ‘қ4(4),4)=1.

Let's define a function рқ‘“(рқ‘қ)=вҲ‘рқ‘–=1рқ‘ҡвҲ’1|рқ‘қрқ‘ңрқ‘ (рқ‘қ,рқ‘Ҙрқ‘–)вҲ’рқ‘қрқ‘ңрқ‘ (рқ‘қ,рқ‘Ҙрқ‘–+1)|, where |рқ‘Јрқ‘Һрқ‘ҷ| is the absolute value of рқ‘Јрқ‘Һрқ‘ҷ. This function means the sum of distances between adjacent elements of рқ‘Ҙ in рқ‘қ.

Your task is to calculate рқ‘“(рқ‘қ1(рқ‘ӣ)),рқ‘“(рқ‘қ2(рқ‘ӣ)),вҖҰ,рқ‘“(рқ‘қрқ‘ӣ(рқ‘ӣ)).

#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
const int maxd = 2e5+10;
vector<int> vis[maxd];
int n,m,a[maxd];
int abs(int x)
{
	if(x < 0) return -x;
	return x;
}
int t[maxd<<1];
void add(int x,int v)
{
	for(;x<maxd*2;x+=(x&-x)) t[x]+=v;
}
int query(int x)
{
	int sum = 0;
	for(;x;x-=(x&-x)) sum += t[x];
	return sum;
}
int main()
{
	// freopen("a.in","r",stdin);
	// freopen("k.out","w",stdout);
	scanf("%d %d",&n,&m);
	for(int i=1;i<=m;i++) 
	{
		scanf("%d",&a[i]);
		vis[a[i]].push_back(i);
	}
	long long ans = 0ll;
	for(int i=2;i<=m;i++) 
	{
		ans += abs(a[i] - a[i-1]);
		int x = a[i], y = a[i-1];
		if(x > y) swap(x,y);
		if(y - x > 1)
		{
			add(x+1,1);
			add(y,-1);
		}
	}
	//printf("%lld ",ans);
	for(int i=1;i<=n;i++)
	{
		long long sum = ans;
		sum -= query(i);
		//printf("%d %lld\n",i,sum);
		for(int j=0;j<vis[i].size();j++)
		{
			int pos = vis[i][j];
			int x = a[pos-1],y = a[pos+1];
			//printf("	%d %d %d\n",pos,x,y);
			if(x != i && x) 
			{
				sum -= abs(i - x);
				if(x<i) sum += x;
				else sum+= 	x-1;
			}
			if(y != i && y)
			{
				sum -= abs(i-y);
				if(y<i) sum+=y;
				else sum+= y-1;
			}
		}
		printf("%lld\n",sum);
	}
 
	return 0;
}
