判断 str 是否在 p 中出现过

#include <cstdio>
using namespace std;
const int maxd = 1e6;
int fail[maxd],ans;
char str[maxd],p[maxd]	// 文本串，匹配串
void getfail()
{
	fail[0] = -1;
	int match = -1;
	for(int i=1;p[i];i++)
	{
		while(match>=0&&p[match+1] != p[i])
			match = fail[match];
		if(p[match+1] == p[i]) match++;
		fail[i] = match;
	}
}
void kmp()
{
	int match = -1;
	for(int i=0;str[i];i++)
	{
		while(match>=0&&p[match+1] != str[i])
			match = fail[match];
		if(p[match+1] == str[i]) 
		{
			match++;
			if(!p[match+1])
			{
				ans = 1;
				return;
			}
		}
	}
}
int main()
{
	fail();
	kmp();
	return 0;
}
