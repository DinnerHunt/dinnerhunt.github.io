using namespace std;
long long n, arr[12], dp[12][12];
void init() {
    dp[0][0] = 1;  // å‰iä¸ªæ•°é¦–ä½ä¸ºjçš„ä¸åŒ…å«62å’Œ4çš„ä¸ªæ•°
    for (int i = 1; i <= 10; i++)
        for (int j = 0; j <= 9; j++)
            for (int k = 0; k <= 9; k++)
                if (j != 4 && !(j == 6 && k == 2))
                    dp[i][j] += dp[i - 1][k];
}
long long solve(long long n) {
    long long len = 0ll, ans = 0ll;
    while (n) arr[++len] = n % 10, n /= 10;
    arr[len + 1] = 0;
    //printf("%lld ",dp[len][0]);
    for (int i = len; i > 0; i--) {
        for (int j = 0; j < arr[i]; j++)
            if (arr[i + 1] != 6 || j != 2)
                ans += dp[i][j];
        if (arr[i] == 4 || (arr[i + 1] == 6 && arr[i] == 2))
            break;

    }
    return ans;
}
