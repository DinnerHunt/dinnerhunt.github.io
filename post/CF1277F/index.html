const ll maxd = 4e5+10;

ll cnt[maxd],n,ans[maxd];

{
    return a.ND > b.ND;

        ll num = min(row,a[i-1].ND);
        while(num--)
        {
            ans[(x-1)*col + y] = a[i-1].first;
            x = (x %row) + 1, y = (y%col) + 1;
            if(ans[(x-1) * col + y]) y = (y%col) + 1;
