const ll N = 2e6+7;
V G[N];
ll f[N],flag = 0;
ll dfs(ll x,ll fa,ll k)
{
    ll cnt = 0;
    for(auto v:G[x])
    {
        if(v == fa) continue;
        cnt += dfs(v,x,k);
    }
    if(cnt % k == 0) return 1;
    if((cnt+1) % k == 0) return 0;
    flag = 0;
    return 0;
}

{
