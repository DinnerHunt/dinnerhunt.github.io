2019-05-04

LCA

#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 3e5+10;
struct
{
    int v,net;
}e[maxd<<1];
int f[maxd],eid,ans,arr[maxd];
void insert(int x,int y)
{
    e[eid].v = y;
    e[eid].net = f[x];
    f[x] = eid++;
}
int d[maxd],p[maxd][30],cnt[maxd],sum[maxd];
void init()
{
    memset(f,-1,sizeof(f));
    memset(d,-1,sizeof(d));
    d[1] = 0;
    eid = 0;
}
void dfs(int x)
{
    for(int i=f[x];i+1;i=e[i].net)
        if(d[e[i].v] == -1)
        {
            d[e[i].v] = d[x] +1;
            p[e[i].v][0] = x;
            dfs(e[i].v);
        }
}
int lca(int x,int y)
{      
    if(d[x] < d[y]) swap(x,y);
    for(int i=25;i>=0;--i)
        if(d[p[x][i]] >= d[y]) x = p[x][i];
    if(x == y) return x;
    for(int i=25;i>=0;i--)
        if(p[x][i]!=p[y][i]) x = p[x][i], y = p[y][i];
    return p[x][0];
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    int n,m,a,b;init();
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",&arr[i]);
    for(int i=1;i<n;i++)
    {
        scanf("%d %d",&a,&b);
        insert(a,b);
        insert(b,a);
    }
    dfs(1);
    for(int j=1;j<=25; j++)
        for(int i=1;i<=n;i++)
            p[i][j] = p[p[i][j-1]][j-1];
    return 0;
}
