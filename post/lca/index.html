V G[N];
struct
{
    ll dp[N][20],d[N];
    void dfs(ll x, ll f) {
        for (auto v:G[x]) {
            if (v == f) continue;
            dp[v][0] = x;
            d[v] = d[x] + 1;
            dfs(v, x);
        }
    }
    void init(ll n)
    {
        d[1] =1;
        dfs(1,1);
        FOR(i,1,19)
            FOR(j,1,n) dp[j][i] = dp[dp[j][i-1]][i-1];
    }
    ll lca(ll x, ll y) {
        if (d[x] < d[y]) swap(x, y);
        _FOR(i, 19, 0) if (d[dp[x][i]] >= d[y]) x = dp[x][i];
        if (x == y) return x;
        _FOR(i, 19, 0) if (dp[x][i] != dp[y][i])x = dp[x][i], y = dp[y][i];
        return dp[x][0];
    }
}a;

    inline ll query(ll x,ll y)

        ll l = st[x],r = st[y];
        if(l>r) swap(l,r);
        int k = log2(r-l+1);
        return val[calc(dp[l][k],dp[r-(1<<k)+1][k])];
