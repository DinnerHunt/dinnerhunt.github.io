LCA
倍增LCA
O(nlogn)O(n log n)O(nlogn) 预处理 ,O(logn)O(log n)O(logn) 查询

V G[N];
struct
{
    ll dp[N][20],d[N];
    void dfs(ll x, ll f) {
        for (auto v:G[x]) {
            if (v == f) continue;
            dp[v][0] = x;
            d[v] = d[x] + 1;
            dfs(v, x);
        }
    }
    void init(ll n)
    {
        d[1] =1;
        dfs(1,1);
        FOR(i,1,19)
            FOR(j,1,n) dp[j][i] = dp[dp[j][i-1]][i-1];
    }
    ll lca(ll x, ll y) {
        if (d[x] < d[y]) swap(x, y);
        _FOR(i, 19, 0) if (d[dp[x][i]] >= d[y]) x = dp[x][i];
        if (x == y) return x;
        _FOR(i, 19, 0) if (dp[x][i] != dp[y][i])x = dp[x][i], y = dp[y][i];
        return dp[x][0];
    }
}a;

struct lca_st
{
    ll dp[N<<1][24],dist[N<<1],val[N<<1],d[N<<1],st[N],cnt;
     ll calc(ll x,ll y){ return d[x] < d[y] ? x : y; }

    //非权版本
     void dfs(ll x,ll fa,ll dd)
    {
        val[++cnt] = x; d[cnt] = dd; st[x] = cnt;
        for(auto v:G[x])
        {
            if(v == fa) continue;
            dfs(v,x,dd+1);
            val[++cnt] = x; d[cnt]= dd;
        }
    }
     void init(ll root,ll n)
    {
        cnt = 0;dist[root] = 0;
        dfs(root,0,1);
        n = 2 * n - 1;
        FOR(i,1,n) dp[i][0] = i;
        for(int j=1;(1<<j)-1<=n;j++)
            for(int i=1;i+(1<<j)-1<=n;i++)
                dp[i][j] = calc(dp[i][j-1],dp[i + (1 << (j - 1))][j-1]);
    }
    inline ll query(ll x,ll y)
    {
        ll l = st[x],r = st[y];
        if(l>r) swap(l,r);
        int k = log2(r-l+1);
        return val[calc(dp[l][k],dp[r-(1<<k)+1][k])];
    }
}O_O;
