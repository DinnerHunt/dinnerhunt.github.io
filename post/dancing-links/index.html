Dancing Links

2019-05-04

用于解决精准覆盖问题 ， 又名双向循环交叉链表

#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
int n,m;
const int maxn = 500;
const int maxd = 500*500;
struct DXL
{
	int n,sz,head;
	int l[maxd],r[maxd],d[maxd],u[maxd];
	int s[maxd];
	int col[maxd],row[maxd]; // 第几列，第几行
	int ans[maxn],ansd;
	void init()
	{
		head = 0;
		ansd = -1;
		for(int i=0;i<=m;i++)
		{
			d[i] = u[i] = i;
			l[i] = i-1; r[i] = i+1;
		}
		r[m] = head; l[head] = m;
		sz = m + 1;
		memset(s,0,sizeof(s));
		//printf("%d ",r[head]);
	}
	void addrow(int x,vector<int> vec)
	{
		int first = sz;
		for(int i=0;i<vec.size();i++)
		{
			int c = vec[i];
			l[sz] = sz - 1; r[sz] = sz+1;
			d[sz] = c , u[sz] = u[c];
			d[u[c]]= sz, u[c] = sz;
			col[sz] = c, row[sz] = x;
			++s[c]; ++ sz;
		}
		r[sz-1] = first;
		l[first] = sz-1;
	}
	void remove(int c)
	{
		r[l[c]] = r[c];
		l[r[c]] = l[c];
		for(int i=d[c];i!=c;i=d[i])
			for(int j=l[i];j!=i;j=l[j])
			{
				d[u[j]] = d[j] , u[d[j]] = u[j];
				s[col[j]]--;
			}
	}
	void restore(int c)
	{
		for(int i=u[c];i!=c;i=u[i])
			for(int j=r[i];j!=i;j=r[j])
			{
				d[u[j]] = j, u[d[j]] = j;
				s[col[j]]++;
			}
		r[l[c]] = c;
		l[r[c]] = c;
	}
	bool dance(int step)
	{
		if(r[head] == head)
		{
			ansd = step;
			return true;
		}
		int c = r[head];
		for(int i=r[head];i!=head;i=r[i])
			if(s[i] < s[c]) c = i;
		remove(c);
		for(int i=d[c];i!=c;i=d[i])
		{
			ans[step] = row[i];
			for(int j=r[i];j!=i;j=r[j]) remove(col[j]);
			if(dance(step+1)) return true;
			for(int j=l[i];j!=i;j=l[j]) restore(col[j]);
		}
		restore(c);
		return false;
	}
	void print()
	{
		if(ansd == -1) printf("No Solution!");
		else for(int i=1;i<ansd;i++)
			printf("%d ",ans[i]);
	}
}Link;
vector<int> vct; 
int main()
{
	// freopen("a.in","r",stdin);
	// freopen("k.out","w",stdout);
	int x;
	scanf("%d %d",&n,&m);
	Link.init();
	for(int i=1;i<=n;i++)
	{
		vct.clear();
		for(int j=1;j<=m;j++)
		{
			scanf("%d",&x);
			if(x) vct.push_back(j);
		}
		Link.addrow(i,vct);
	}
	Link.dance(1);
	Link.print();
	return 0;
}
