「FWT」多项式位运算卷积

2019-09-09 #FWT

给定两个长度为2n2^n2n的两个序列A,BA,BA,B,设Ci=∑j∗kAjAkC_i = \sum_{j*k}A_jA_kCi​=∑j∗k​Aj​Ak​,分别是当∗*∗ 为or,and,xor时，结果mod998244353mod 998244353mod998244353

#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
const int maxd = 2<<18;
const int mod = 998244353;
int n;
long long a[maxd],b[maxd],c[maxd],inv2;
long long pow(int x,int y)
{
    long long ans = 1,now = x;
    while(y)
    {
        if(y&1) ans = (now * ans) % mod;
        now = ( now * now) %mod;
        y >>=1;
    }
    return ans;
}
void FWT_or(long long *a,int opt)
{
    for(int i=1;i<n;i<<=1)
        for(int p=i<<1,j=0;j<n;j+=p)
            for(int k=0;k<i;++k)
                if(opt==1)a[i+j+k]=(a[j+k]+a[i+j+k])%mod;
                else a[i+j+k]=(a[i+j+k]+mod-a[j+k])%mod;
}
void FWT_and(long long *a,int opt)
{
    for(int i=1;i<n;i<<=1)
        for(int p=i<<1,j=0;j<n;j+=p)
            for(int k=0;k<i;++k)
                if(opt==1)a[j+k]=(a[j+k]+a[i+j+k])%mod;
                else a[j+k]=(a[j+k]+mod-a[i+j+k])%mod;
}
void FWT_xor(long long *a,int opt)
{
    for(int i=1;i<n;i<<=1)
        for(int p=i<<1,j=0;j<n;j+=p)
            for(int k=0;k<i;++k)
            {
                int X=a[j+k],Y=a[i+j+k];
                a[j+k]=(X+Y)%mod;a[i+j+k]=(X+mod-Y)%mod;
                if(opt==-1)a[j+k]=1ll*a[j+k]*inv2%mod,a[i+j+k]=1ll*a[i+j+k]*inv2%mod;
            }
}
int main() 
{

    scanf("%d",&n); n = 1<<n;inv2 = pow(2,mod-2);

    for(int i=0;i<n;i++) scanf("%lld",&a[i]);
    for(int i=0;i<n;i++) scanf("%lld",&b[i]);
    FWT_or(a,1);FWT_or(b,1);
    for(int i=0;i<n;i++) c[i] = (1ll * a[i] * b[i] %mod);
    FWT_or(c,-1);FWT_or(a,-1);FWT_or(b,-1);
    for(int i=0;i<n;i++) printf("%lld ",c[i]);
    putchar(10);

    FWT_and(a,1);FWT_and(b,1);
    for(int i=0;i<n;i++) c[i] = (1ll * a[i] * b[i] %mod);
    FWT_and(c,-1);FWT_and(a,-1);FWT_and(b,-1);
    for(int i=0;i<n;i++) printf("%lld ",c[i]);
    putchar(10);

    FWT_xor(a,1);FWT_xor(b,1);
    for(int i=0;i<n;i++) c[i] = (1ll * a[i] * b[i] %mod);
    FWT_xor(c,-1);FWT_xor(a,-1);FWT_xor(b,-1);
    for(int i=0;i<n;i++) printf("%lld ",c[i]);
}
