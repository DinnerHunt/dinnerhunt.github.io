普通平衡树

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <ctime>
using namespace std;
const int maxd = 5e5+10;
inline int read()
{
    int x=0,f=1;char c = getchar();
    while(c>'9'||c<'0'){if(c=='-')f=-1;c=getchar();}
    while(c>='0'&&c<='9'){x=x*10+c-'0';c=getchar();}
    return x*f;
}
struct
{
    int ch[maxd][2],size[maxd],val[maxd],fix[maxd];
    int root,cnt,x,y,z;
    void init()
    {
        srand((unsigned)time(NULL));
        root = x = y = z = cnt = 0;
    }
    inline void pushup(int x)
    {
        size[x] = 1+size[ch[x][0]] + size[ch[x][1]];
    }
    inline int new_node(int x)
    {
        size[++cnt] = 1;
        val[cnt] = x;
        fix[cnt] = rand();
        return cnt;
    }
    int merge(int a,int b)  // 这两棵树合并
    {
        if(!a||!b) return a+b; // 如果只有一棵子树
        if(fix[a] < fix[b]){ch[a][1]=merge(ch[a][1],b);pushup(a);return a;}
        ch[b][0] = merge(a,ch[b][0]);pushup(b); return b;
    }
    void split(int now,int k,int &x,int &y) // 将now为根的树，以权值k为界限，划分成两棵树
    {
        if(!now) x=y=0; // 如果到达叶子节点
        else
        {
            if(val[now] <= k)  
            {
                x = now;
                split(ch[now][1],k,ch[now][1],y);
            }
            else
            {
                y = now;
                split(ch[now][0],k,x,ch[now][0]);
            }
            pushup(now);
        }
    }
    int kth(int now,int k)
    {
        while(1)
        {
            if(k<=size[ch[now][0]]) now = ch[now][0];
            else if(k==size[ch[now][0]]+1) return now;
            else k-=size[ch[now][0]]+1,now=ch[now][1];
        }
        return 0;
    }
    void insert(int v)
    {
        split(root,v,x,y);
        root = merge(merge(x,new_node(v)),y);
    }
    void del(int v)
    {
        split(root,v,x,z); // 将比a小的放x,比a大的放z
        split(x,v-1,x,y);  // 比a-1小的放x，比a-1大的放y,这时a是y的根节点
        y = merge(ch[y][0],ch[y][1]); // 去根
        root = merge(merge(x,y),z); 
    }
    int rank(int v)
    {
        split(root,v-1,x,y);
        int k = size[x]+1;
        root = merge(x,y);
        return k;
    }
    int pre(int v)
    {
        split(root,v-1,x,y);
        int k =val[kth(x,size[x])];
        root = merge(x,y);
        return k;
    }
    int net(int v)
    {
        split(root,v,x,y);
        int k = val[kth(y,1)];
        root = merge(x,y);
        return k;
    }
}Treap;
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    int t,opt,a;t=read();Treap.init();
    while(t--)
    {
        opt=read();a=read();
        if(opt == 1) Treap.insert(a);
        if(opt == 2) Treap.del(a);
        if(opt == 3) printf("%d\n",Treap.rank(a));
        if(opt == 4) printf("%d\n",Treap.val[Treap.kth(Treap.root,a)]);
        if(opt == 5) printf("%d\n",Treap.pre(a));
        if(opt == 6) printf("%d\n",Treap.net(a));
    }
    return 0;
}

#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
const int maxd = 5e5+10;
inline int read()
{
    int x=0,f=1;char c = getchar();
    while(c>'9'||c<'0'){if(c=='-')f=-1;c=getchar();}
    while(c>='0'&&c<='9'){x=x*10+c-'0';c=getchar();}
    return x*f;
}
struct
{
    int ch[maxd][2],size[maxd],val[maxd],fix[maxd];
    int root,cnt,x,y,z;
    bool lazy[maxd];
    void init()
    {
        srand((unsigned)time(NULL));
        root = x = y = z = cnt = 0;
    }
    inline void pushup(int x)
    {
        size[x] = 1+size[ch[x][0]] + size[ch[x][1]];
    }
    inline void pushdown(int x)
    {
        if(!lazy[x]) return;
        swap(ch[x][0],ch[x][1]);
        if(ch[x][0]) lazy[ch[x][0]]^=1;
        if(ch[x][1]) lazy[ch[x][1]]^=1;
        lazy[x]=0;
    }
    inline int new_node(int x)
    {
        size[++cnt] = 1;
        val[cnt] = x;
        fix[cnt] = rand();
        return cnt;
    }
    int merge(int a,int b)  // 这两棵树合并
    {
        if(!a||!b) return a+b; // 如果只有一棵子树
        if(fix[a] < fix[b])
        {
            pushdown(a);
            ch[a][1]=merge(ch[a][1],b);
            pushup(a);
            return a;
        }
        else{
            pushdown(b);
            ch[b][0] = merge(a,ch[b][0]);
            pushup(b);
            return b;
        }
    }

    // void split(int now,int k,int &x,int &y) // 以子节点个数为分界 // 区间反转操作需要

    void split(int now,int k,int &x,int &y) // 将now为根的树，以权值k为界限，划分成两棵树
    {
        if(!now) x=y=0; // 如果到达叶子节点
        else
        {
            pushdown(now);
            if(val[now] <= k)  
            {
                x = now;
                split(ch[now][1],k,ch[now][1],y);
            }
            else
            {
                y = now;
                split(ch[now][0],k,x,ch[now][0]);
            }
            pushup(now);
        }
    }
    int kth(int now,int k)  //第k大
    {
        while(1)
        {
            if(k<=size[ch[now][0]]) now = ch[now][0];
            else if(k==size[ch[now][0]]+1) return now;
            else k-=size[ch[now][0]]+1,now=ch[now][1];
        }
        return 0;
    }
    void coutt(int now)
    {
        if(!now) return;
        pushdown(now);
        coutt(ch[now][0]);
        printf("%d ",val[now]);
        coutt(ch[now][1]);
    }

    void insert(int v)  //插入
    {
        split(root,v,x,y);
        root = merge(merge(x,new_node(v)),y);
    }
    void del(int v) //删除
    {
        split(root,v,x,z); // 将比a小的放x,比a大的放z
        split(x,v-1,x,y);  // 比a-1小的放x，比a-1大的放y,这时a是y的根节点
        y = merge(ch[y][0],ch[y][1]); // 去根
        root = merge(merge(x,y),z); 
    }
    int rank(int v) //排名
    {
        split(root,v-1,x,y);
        int k = size[x]+1;
        root = merge(x,y);
        return k;
    }
    int pre(int v)  //前驱
    {
        split(root,v-1,x,y);
        int k =val[kth(x,size[x])];
        root = merge(x,y);
        return k;
    }
    int net(int v)  //后继
    {
        split(root,v,x,y);
        int k = val[kth(y,1)];
        root = merge(x,y);
        return k;
    }
    void rotate(int a,int b)
    {
        split(root,a-1,x,y);
        split(y,b-a+1,y,z); //将要旋转的部分移动到y节点
        lazy[y]^=1;
        root = merge(x,merge(y,z));
    }
}Treap;
