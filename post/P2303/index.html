#include <cstdio>
#include <algorithm>
using namespace std;
long long factor[1600],n,m;
long long ans;
long long phi(long long n)
{
    long long ans = n;
    for(long long i=2;i*i<=n;i++)
        if(n%i==0)
        {
            ans = ans / i * (i-1);
            while(n%i == 0) n/=i;
        } 
    if(n>1) ans = ans /n *( n-1 );
    return ans;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%lld",&n);
    for(long long i=1;i*i<=n;i++)
    {
        if(n%i == 0)
        {
            factor[++m] = i;
            if(i!=n/i) factor[++m] = n/i;
        }
    }
    for(long long i=1;i<=m;i++)
        ans += factor[i] * phi(n/factor[i]);
    printf("%lld",ans);                               
    return 0;
}
