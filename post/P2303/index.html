「SDOI2012」 Longge的问题

2019-05-05 数论

Longge的数学成绩非常好，并且他非常乐于挑战高难度的数学问题。现在问题来了：给定一个整数n，你需要求出∑i=1ngcd(i,n)\sum_{i=1}^n gcd(i,n)∑i=1n​gcd(i,n)

[SDOI2012] Longge的问题

题解

每个gcd(i,n) 一定是n的约数

∑i=1ngcd(i,n)=∑d∣nd∑i=1ngcd(id,nd)) \sum_{i=1}^ngcd(i,n) = \sum_{d|n}d\sum_{i=1}^ngcd(\frac{i}{d},\frac{n}{d})) i=1∑n​gcd(i,n)=d∣n∑​di=1∑n​gcd(di​,dn​))

=∑d∣nd∑i=1ndgcd(i,nd)=1 = \sum_{d|n}d\sum_{i=1}^{\frac{n}{d}}gcd(i,\frac{n}{d})=1 =d∣n∑​di=1∑dn​​gcd(i,dn​)=1

=∑d∣ndφ(nd)= \sum_{d|n}d \varphi(\frac{n}{d}) =d∣n∑​dφ(dn​)

枚举n的约数即可

代码

#include <cstdio>
#include <algorithm>
using namespace std;
long long factor[1600],n,m;
long long ans;
long long phi(long long n)
{
    long long ans = n;
    for(long long i=2;i*i<=n;i++)
        if(n%i==0)
        {
            ans = ans / i * (i-1);
            while(n%i == 0) n/=i;
        } 
    if(n>1) ans = ans /n *( n-1 );
    return ans;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%lld",&n);
    for(long long i=1;i*i<=n;i++)
    {
        if(n%i == 0)
        {
            factor[++m] = i;
            if(i!=n/i) factor[++m] = n/i;
        }
    }
    for(long long i=1;i<=m;i++)
        ans += factor[i] * phi(n/factor[i]);
    printf("%lld",ans);                               
    return 0;
}
