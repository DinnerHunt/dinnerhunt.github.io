[LOJ 10153] 二叉苹果树

题意

有一棵二叉苹果树，如果数字有分叉，一定是分两叉，即没有只有一个儿子的节点。这棵树共 N 个节点，标号 1 至 N，树根编号一定为 1。

我们用一根树枝两端连接的节点编号描述一根树枝的位置。一棵有四根树枝的苹果树，因为树枝太多了，需要剪枝。但是一些树枝上长有苹果，给定需要保留的树枝数量，求最多能留住多少苹果。

题解

树形DP中的背包问题
具体看代码注释

代码
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 105;
int dp[maxd][maxd],vis[maxd],arr[maxd];
int n,m,a,b,c;
struct
{
	struct node
	{
		int v,net,w;
	}e[maxd];
	int f[maxd],eid;
	void init()
	{
		memset(f,-1,sizeof(f));
		eid=0;
	}
	void insert(int a,int b,int c)
	{
		e[eid].v = b;
		e[eid].net = f[a];
		e[eid].w = c;
		f[a] = eid++;
	}
}G;
int dfs(int x)
{
	if(G.f[x] == -1) return 0;
	vis[x] = 1;
	int sum = 0;
	for(int i=G.f[x];i+1;i=G.e[i].net)
	{
		int v = G.e[i].v;
		if(vis[v]) continue;
		int t = dfs(v)+1;
		for(int j=0;j<=sum;j++) arr[j] = dp[x][j];
		for(int j=0;j<=sum;j++)	
			for(int k=1;k<=t&&j+k<=m;k++)
				dp[x][j+k] = max(dp[x][j+k],arr[j]+dp[v][k-1]+G.e[i].w);
		sum += t;
	}
	return sum;
}
int main()
{
	// freopen("a.in","r",stdin);
	// freopen("k.out","w",stdout);
	G.init();
	scanf("%d %d",&n,&m);
	for(int i=1;i<n;i++)
	{
		scanf("%d %d %d",&a,&b,&c);
		G.insert(a,b,c);
		G.insert(b,a,c);
	}
	dfs(1);
	printf("%d",dp[1][m]);
	return 0;
}
