#include <cstdio>
#include <vector>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 2e5+10;
int d[maxd],vis[maxd],cnt[maxd],n,m;
typedef pair<int,int> P;
vector<P> e;
vector<int> G[maxd];
int main()
{
//    freopen("a.in","r",stdin);
//    freopen("k.out","w",stdout);
    while(scanf("%d %d",&n,&m)!=EOF)
    {
        int ans = 0ll; // СИЅтЁЃуј»ТЋ░
        for (int i = 1; i <= m; i++)
        {
            int u,v;scanf("%d %d", &u, &v);
            e.push_back(P(u,v));
            ++d[u], ++d[v];
        }
        for (int i = 0; i < m; i++)
        {
            int u = e[i].first, v = e[i].second;
            if (d[u] < d[v] || (d[u] == d[v] && u > v)) swap(u, v);
            G[u].push_back(v);
        }
        for (int i = 1; i <= n; i++)
        {
            for (int u:G[i]) vis[u] = i;
            for (int u:G[i])
                for (int v:G[u])
                    if (vis[v] == i) ans++;
        }
        printf("%d\n", ans);
    }
    return 0;
}
