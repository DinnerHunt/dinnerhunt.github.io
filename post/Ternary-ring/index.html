无向图三元环计数

O(msqrt(m))O(m sqrt(m))O(msqrt(m))

ll solve(ll n,ll m)
{
    ll a,b;
    FOR(i,1,n)
    {
        RLL2(a,b);e.pb(P(a,b));
        ++d[a],++d[b];
    }
    FOR(i,1,m)
    {
        ll u = e[i].first,v = e[i].second;
        if(d[u] < d[v] || (d[u] == d[v] && u > v)) swap(u,v);
        G[u].pb(v);
    }
    ll ans = 0;
    FOR(i,1,n)
    {
        for(auto u:G[i]) vis[u] = i;
        for(auto u:G[i]) for(auto v:G[u]) if(vis[v] == i) ans++;
    }
    return ans;
}
