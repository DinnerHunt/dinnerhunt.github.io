拉格朗日插值

2019-05-04

P4781 拉格朗日插值

f(k)=∑i=0nyi∏i!=jk−x[j]x[i]−x[j] f(k) = \sum_{i=0}^n y_i \prod_{i!=j}{\frac{k-x[j]}{x[i] - x[j]}} f(k)=​i=0​∑​n​​y​i​​​i!=j​∏​​​x[i]−x[j]​​k−x[j]​​

根据公式，我们可以在O(n^2)时间内根据n个坐标点求出f(k)的值

#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
const int maxd = 2005;
const ll mod = 998244353;
ll x[maxd],y[maxd],n,k,ans;
long long pow(long long x,long long y)
{
    long long ans = 1;
    while(y)
    {
        if(y&1) ans=ans * x %mod;
        x = x*x%mod;
        y>>=1;
    }
    return ans;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%lld %lld",&n,&k);
    for(int i=1;i<=n;i++) scanf("%lld %lld",&x[i],&y[i]);
    for(int i=1;i<=n;i++)
    {
        ll sum=1;
        for(int j=1;j<=n;j++)
            if(j!=i) sum = sum*(x[i]+mod-x[j]) %mod;
        sum = pow(sum,mod-2); // 费马小定理求逆元
        for(int j=1;j<=n;j++)
            if(i!=j) sum = sum*(k+mod - x[j])%mod;
        sum = sum * y[i] %mod;
        ans = (ans+sum)%mod;
    }
    printf("%lld",ans);
    return 0;
}
