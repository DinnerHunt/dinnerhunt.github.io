P4781 拉格朗日插值

给定n个点，请你确定这个多项式，并将k代入求值

求出的值对998244353998244353取模

#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
const int maxd = 2005;
const ll mod = 998244353;
ll x[maxd],y[maxd],n,k,ans;
long long pow(long long x,long long y)
{
    long long ans = 1;
    while(y)
    {
        if(y&1) ans=ans * x %mod;
        x = x*x%mod;
        y>>=1;
    }
    return ans;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%lld %lld",&n,&k);
    for(int i=1;i<=n;i++) scanf("%lld %lld",&x[i],&y[i]);
    for(int i=1;i<=n;i++)
    {
        ll sum=1;
        for(int j=1;j<=n;j++)
            if(j!=i) sum = sum*(x[i]+mod-x[j]) %mod;
        sum = pow(sum,mod-2); // 费马小定理求逆元
        for(int j=1;j<=n;j++)
            if(i!=j) sum = sum*(k+mod - x[j])%mod;
        sum = sum * y[i] %mod;
        ans = (ans+sum)%mod;
    }
    printf("%lld",ans);
    return 0;
}
