#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#define pi 3.1415926
#define INF 55000
#define eps 1e-8
using namespace std;
struct point {
    double x,y;
    point operator + (const point& k1)const{return (point){x+k1.x,y+k1.y};}
    point operator - (const point& k1)const{return (point){x-k1.x,y-k1.y};}
    point operator * (const double& k1)const{return (point){x*k1,y*k1};}
    point operator / (const double& k1)const{return (point){x/k1,y/k1};}
    
    friend bool operator <(point a, point b){
        if(fabs(a.x-b.x) > eps) return a.x < b.x;
        return a.y < b.y;
    }
    friend bool operator <= (point a, point b){
        if(fabs(a.y-b.y) > eps) return a.y < b.y;
        return a.x < b.x;
    }
    friend bool operator ==(point a, point b){
        return (a.x==b.x && a.y==b.y);
    }

}p[INF];
point stk[INF];
// 内积 
double dot(point a,point b) {
    return a.x*b.x+a.y*b.y;
}
// 叉乘  
point cross1(point a,point b) {
    return (point){a.x-b.x,a.y-b.y};
}
double crossProduct(point a, point b, point c) {
    point ai = cross1(a,b);
    point bi = cross1(b,c);
    return (double)ai.x*bi.y - ai.y*bi.x;
}
double cross(point a,point b) {
    return a.x*b.y-a.y*b.x;
} 
// 面积
double area(point a,point b,point c){return fabs(cross(a-b,a-c));}
// 处理-0精度
double F(int a) {
    return a<0?a=-a:0;
}
// 相交
point intersect(point k1,point v,point k2,point w){
    double x=cross(w,k1-k2)/cross(v,w);
    return k1+v*x;
}
// 映射 
point project(point p,point k1,point v){
    double x=dot(v,p-k1)/dot(v,v);
    return k1+v*x;
} 
// 两点间距离 
double dist(point a, point b) {
    return sqrt((a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y)+0.0);
}
// 计算边长
double len(point a){return sqrt(a.x*a.x+a.y*a.y);}
// 点到直线 
double distanceToLine(point p,point k1,point v) {
    return fabs(cross(v,p-k1)) / len(v);
}
// 坐标比较 
int cmp(point a, point b) {
    if (a.y == b.y) return a.x < b.x;
    return a.y < b.y;
}
//凸包 
int Graham(int n) {
    sort(p+1, p+1+n);
    int top=0;

    for (int i=1 ;i<=n ; i++) {
        while (top>1 && cross(stk[top]-stk[top-1],p[i]-stk[top])<=0)    top--;
        stk[++top] = p[i];
    }

    int qaq = top;
    for(int i=n-1; i>=1; i--) {
        while(top>qaq && cross(stk[top]-stk[top-1],p[i]-stk[top])<=0) top--;
        stk[++top] = p[i];
    }
    return top; 
}
// 旋转卡壳
point points[5];
void solve(int top)
{
    double ans = 1e18;
    int a1,a2,a3;a1=a2=a3=2;
    stk[top+1]=stk[1];
    for(int i=2; i<=top; i++) {
        // 找到最上的点 
        while(cross(stk[a1]-stk[i],stk[i-1]-stk[i]) <= cross(stk[a1+1]-stk[i],stk[i-1]-stk[i]))     a1=a1%top+1;
        // 找到最右的点 
        while(dot(stk[a2]-stk[i],stk[i-1]-stk[i]) >= dot(stk[a2+1]-stk[i],stk[i-1]-stk[i]))     a2=a2%top+1;
        if(i==2)    a3=a1;
        // 找到最左的点 
        while(dot(stk[a3]-stk[i-1],stk[i]-stk[i-1]) >= dot(stk[a3+1]-stk[i-1],stk[i]-stk[i-1])) a3=a3%top+1;
        double dis=len(stk[i]-stk[i-1]);
        double L=dot(stk[a3]-stk[i],stk[i-1]-stk[i])/dis;F(L);
        double R=dot(stk[a2]-stk[i-1],stk[i]-stk[i-1])/dis;F(R);
        double H=area(stk[i],stk[i-1],stk[a1]) / len(stk[i]-stk[i-1]);F(H);
        double tmp=(L+R-dis)*H;
        if(tmp<ans){
            ans=tmp;
            points[1]=stk[i]-(stk[i]-stk[i-1])*(L/dis);
            points[2]=points[1]+(stk[i]-stk[i-1])*((R+L-dis)/dis);
            points[3]=points[2]+(stk[a2]-points[2])*(H/len(stk[a2]-points[2]));
            points[4]=points[3]+(stk[i-1]-stk[i])*((R+L-dis)/dis); 
        }
    }
    printf("%.5lf\n",ans);
    int st=0;points[0].y=1e9;
    for(int i=1;i<=4;i++)if(points[i].y<points[st].y||(points[i].y==points[st].y&&points[i].x<points[st].x))st=i;
    if(fabs(points[st].x)<=1e-8)points[st].x=0;
    if(fabs(points[st].y)<=1e-8)points[st].y=0;
    printf("%.5lf %.5lf\n",points[st].x,points[st].y);
    for(int i=0;i<3;i++){
        if(fabs(points[(i+st)%4+1].x)<=1e-8)points[(i+st)%4+1].x=0;
        if(fabs(points[(i+st)%4+1].y)<=1e-8)points[(i+st)%4+1].y=0;
        printf("%.5lf %.5lf\n",points[(i+st)%4+1].x,points[(i+st)%4+1].y);
    }
}
int main()
{
    int n,d;
    cin >> n;
    d *= 2;
    for(int i=1; i<=n; i++) {
        cin >> p[i].x >> p[i].y;
    }
    int len = Graham(n);
    solve(len);
    return 0;
}
