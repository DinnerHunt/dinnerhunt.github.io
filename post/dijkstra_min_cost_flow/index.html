#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<queue>
#include<string>
#include<fstream>
#include<vector>
#include<stack>
#include <map>
using namespace std;
typedef long long ll;
const int inf = 2e9;
const int maxd = 5e3+10;
struct Edge
{
    int to, rev;
    int flow, cost;;
};
int num;
vector<Edge> edge[maxd];
int h[maxd];
int dis[maxd];
int pre_node[maxd], pre_edge[maxd];

void add(int u, int v, int flow, int cost)
{
    edge[u].push_back((Edge){ v,(int)edge[v].size(),flow,cost });
    edge[v].push_back((Edge){ u,(int)edge[u].size() - 1,0,-cost });
}
typedef pair<int,int> P;
void min_cost_flow(int s, int t, int& min_cost, int& max_flow)
{
    fill(h + 1, h + 1 + num, 0);
    min_cost = max_flow = 0;
    int tot = inf;
    while (tot > 0)
    {
        priority_queue<P, vector<P>, greater<P> > q;
        fill(dis, dis + 1 +  num, inf);
        dis[s] = 0;q.push(P(0,s));
        while (!q.empty())
        {
            int u = q.top().second;
            int dist = q.top().first;
            q.pop();
            if (dis[u] < dist)continue;
            for (int i = 0;i < edge[u].size(); i++)
            {
                Edge &e = edge[u][i];
                if (edge[u][i].flow > 0 && dis[e.to] > dis[u] + e.cost + h[u] - h[e.to])
                {
                    dis[e.to] = dis[u] +e.cost + h[u] - h[e.to];
                    pre_node[e.to] = u;
                    pre_edge[e.to] = i;
                    q.push(P(dis[e.to],e.to ));
                }
            }
        }
        if (dis[t] == inf)break;
        //这里要注意点的范围
        for (int i = 0;i <= num;i++) h[i] += dis[i];
        int flow = tot;
        for (int i = t; i != s; i = pre_node[i])
            flow = min(flow, edge[pre_node[i]][pre_edge[i]].flow);
        for (int i = t; i != s; i = pre_node[i])
        {
            Edge& e = edge[pre_node[i]][pre_edge[i]];
            e.flow -= flow;
            edge[i][e.rev].flow += flow;
        }
        tot -= flow;
        max_flow += flow;
        min_cost += flow * h[t];
    }
}
void init()
{
    for(int i=1;i<=num;i++) edge[i].clear();
}
int main()
{
    int s, t ,n,m, T;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&n,&m);s = 0, t= n+1, num= t; init();
        int min_cost, max_flow;
        min_cost_flow(s, t, min_cost, max_flow);
        printf("%d\n", min_cost);

    }
    return 0;
}
