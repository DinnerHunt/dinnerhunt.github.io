const int N = 5e3+10;
V G[N];
ll a[N],dp[N][N],s[N],sz[N],n,m;
void dfs(ll x,ll fa)
{
    sz[x] = 1;
    dp[x][1] = dp[x][0] = 0;
    for(auto v: G[x])
    {
        if(v == fa) continue;
        dfs(v,x);
        ll now = 1e18;
        FOR(i,0,sz[v])
        {
            if(i >= m) break;
            now = min(now,dp[v][i]);
        }
        dp[x][0] += now;
        FOR(i,1,sz[x]+sz[v]) s[i] = 1e18;
        FOR(i,1,sz[x])if(i < m)
            FOR(j,0,sz[v]) if(i+j<m)
                s[max(i,j+1)] = min(s[max(i,j+1)],dp[x][i] + dp[v][j]);
        sz[x]+= sz[v];
        FOR(i,1,sz[x]) dp[x][i] = s[i];
    }
    dp[x][0] += a[x];
}
int main()
{
//    __IN;__OUT;
    RLL2(n,m);
    FOR(i,1,n) RLL(a[i]);
    FOR(i,1,n-1)
    {
        ll x,y;RLL2(x,y);
        G[x].pb(y); G[y].pb(x);
    }
    MM(dp,233);
    dfs(1,1);
    ll ans = dp[1][0];
    FOR(i,1,m-1)  ans = min(ans,dp[1][i]);
    PLN(ans);
    return 0;
}
