2019-05-04

UOJ 欧拉回路

#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 4e5 + 10;
struct node {
    int v, net;
} e[maxd];
int f[maxd], eid;
int t, n, m, a, check[maxd];
bool use[maxd];
int ans[maxd], top;
void init() {
    memset(f, -1, sizeof(f));
    eid = 1;
}
void insert(int x, int y) {
    e[eid].v = y;
    e[eid].net = f[x];
    f[x] = eid++;
}
void solve(int x) {
    for (int i = f[x]; i + 1; i = f[x]) {
        f[x] = e[i].net;  //删边以防超时
        if (t == 1) {
            if (!use[(i + 1) >> 1]) {
                use[(i + 1) >> 1] = 1;
                solve(e[i].v);
                ans[++top] = ((i + 1) >> 1) * (i % 2 == 0 ? -1 : 1);
            }
        } else {
            solve(e[i].v);
            ans[++top] = i;
        }
    }
}
int main() {
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    int a, b;
    init();
    scanf("%d %d %d", &t, &n, &m);
    for (int i = 1; i <= m; i++) {
        scanf("%d %d", &a, &b);
        insert(a, b);
        check[a]++;
        if (t == 1) {
            insert(b, a);
            check[b]++;
        } else
            check[b]--;
    }
    for (int i = 1; i <= n; i++) {
        if (t == 1) {
            if (check[i] & 1) {
                printf("NO");
                return 0;
            }
        } else if (check[i]) {
            printf("NO");
            return 0;
        }
    }
    solve(a);
    if (top < m) {
        printf("NO");
        return 0;
    }
    printf("YES\n");
    for (int i = top; i; i--) printf("%d ", ans[i]);
    puts("");
    return 0;
}
