#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 1e5+100;
int n,m,root,x,y;
long long arr[maxd];
int fa[maxd],d[maxd],siz[maxd],son[maxd],rk[maxd],id[maxd],top[maxd];
struct  //  еӯҳеӣҫ
{
    struct
    {
        int v,net;
    }e[maxd<<1];
    int f[maxd],eid;
    void init()
    {
        memset(f,-1,sizeof(f));
        eid = 0;
    }
    void insert(int x,int y)
    {
        e[eid].v = y;
        e[eid].net = f[x];
        f[x] = eid++;
    }
}G;
struct  //  зәҝж®өж ‘
{
    struct
    {
        int l,r,ls,rs;
        long long sum,lazy;
    }tree[maxd<<2];
    int cnt;
    void init()
    {
        cnt = 0;
    }
    int len(int x)
    {
        return tree[x].r - tree[x].l + 1;
    }
    void pushup(int x)
    {
        int ls = tree[x].ls, rs = tree[x].rs;
        tree[x].sum = tree[ls].sum + tree[rs].sum;
    }
    void build(int l,int r,int x)
    {
        if(l==r)
        {
            tree[x].sum = arr[rk[l]];
            tree[x].l = tree[x].r = l;
            return ;
        }
        int mid = (l+r) >> 1;
        tree[x].ls = ++cnt,tree[x].rs = ++cnt;
        tree[x].l = l , tree[x].r = r;
        build(l,mid,tree[x].ls);
        build(mid+1,r,tree[x].rs);
        pushup(x);
    }
    void pushdown(int x)
    {
        if(tree[x].lazy==0) return;
        int ls = tree[x].ls,rs = tree[x].rs;
        long long lz= tree[x].lazy;
        (tree[ls].sum += lz*len(ls));
        (tree[rs].sum += lz*len(rs));
        (tree[ls].lazy += lz);
        (tree[rs].lazy += lz);
        tree[x].lazy = 0;
    }
    void update(int l,int r,long long v,int x)
    {
        if(l<=tree[x].l && r>=tree[x].r)
        {
            (tree[x].sum += v*len(x));
            (tree[x].lazy += v);
            return;
        }
        pushdown(x);
        int mid = (tree[x].l+tree[x].r) >> 1;
        if(l <= mid) update(l,r,v,tree[x].ls);
        if(r>mid)    update(l,r,v,tree[x].rs);
        pushup(x);
    }
    long long query(int l,int r,int x)
    {
        if(l<=tree[x].l && r >= tree[x].r)
            return tree[x].sum;
        pushdown(x);
        int mid = (tree[x].l+tree[x].r) >> 1;
        long long ans = 0;
        if(l<=mid) ans += query(l,r,tree[x].ls);
        if(r>mid)  ans += query(l,r,tree[x].rs);
        return ans;
    }
}Tree;
struct
{
    int cnt;
    void init(int root)
    {
        cnt = 0;
        dfs(root);
        dfs1(root,root);
    }
    void dfs(int x)
    {
        siz[x] = 1;d[x] = d[fa[x]]+1;
        for(int i=G.f[x];i+1;i=G.e[i].net)
        {
            int v = G.e[i].v;
            if(v==fa[x]) continue;
            fa[v] = x;
            dfs(v);
            siz[x] += siz[v];
            if(siz[v] > siz[son[x]])
                son[x] = v;
        }
    }
    void dfs1(int x,int y)                  //  йў„еӨ„зҗҶ
    {
        top[x]=y; id[x]=++cnt; rk[cnt] = x;
        if(son[x]) dfs1(son[x],y);
        for(int i=G.f[x];i+1;i=G.e[i].net)
        {
            int v = G.e[i].v;
            if(v!=son[x]&&v!=fa[x])dfs1(v,v);
        }
    }
    long long sum_lca(int x,int y)          // жұӮи·Ҝеҫ„зҡ„е’Ң
    {
        long long ans = 0;
        int fx = top[x], fy = top[y];
        while(fx!=fy)
        {
            if(d[fx] >= d[fy])
            {
                (ans += Tree.query(id[fx],id[x],0));
                x = fa[fx], fx = top[x];
            }
            else
            {
                (ans += Tree.query(id[fy],id[y],0));
                y = fa[fy], fy = top[y];
            }
        }
        if(id[x] <= id[y]) (ans += Tree.query(id[x],id[y],0));
        else (ans += Tree.query(id[y],id[x],0));
        return ans;
    }
    void change(int x,int y,long long v)    // ж”№еҸҳдёҖжқЎи·Ҝеҫ„зҡ„еҖј
    {   
        int fx = top[x], fy = top[y];
        while(fx!=fy)
        {
            if(d[fx]>=d[fy])
            {
                Tree.update(id[fx],id[x],v,0);
                x = fa[fx], fx = top[x];
            }
            else
            {
                Tree.update(id[fy],id[y],v,0);
                y = fa[fy], fy = top[y];
            }
        }
        //printf("%d %d %lld\n",id[x],id[y],v);
        if(id[x] <= id[y]) Tree.update(id[x],id[y],v,0);
        else Tree.update(id[y],id[x],v,0);
    }
    int lca(int x,int y)   // LCA
    {
        while(top[x]!=top[y])
        {
            if(d[top[x]]>d[top[y]])swap(x,y);
            y=fa[top[y]];
        }
        return d[x]>d[y]?y:x;
    }
    int find(int x,int y)   //иҝ”еӣһxеҲ°yзӮ№пјҢxзӮ№зӣёйӮ»зҡ„дҪҚзҪ®
    {
        while(top[x]!=top[y])
        {
            if(d[top[x]]<d[top[y]]) swap(x,y);
            if(fa[top[x]]==y) return top[x];
            x=fa[top[x]];
        }
        if(d[x]<d[y])swap(x,y);
        return son[y];
    }
    void Tree_update(int x,int y)       // жҚўж №ж“ҚдҪң
    {   
        if(root == x) return Tree.update(1,n,y,0);
        int k = lca(x,root);
        if(k != x) return Tree.update(id[x],id[x]+siz[x]-1,y,0);
        int son = find(x,root);
        Tree.update(1,n,y,0);
        Tree.update(id[son],id[son]+siz[son]-1,-y,0);
    }
    long long Tree_query(int x)
    {
        if(root == x) return Tree.query(1,n,0);
        int k = lca(x,root);
        if(k != x) return Tree.query(id[x],id[x]+siz[x]-1,0);
        int son = find(x,root);
        return Tree.query(1,n,0)-Tree.query(id[son],id[son]+siz[son]-1,0);
    }
}Tree_link;
