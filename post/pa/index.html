回文自动机

2019-08-29 回文自动机

[APIO2014]回文串

求字符串中，回文子串长度与该串出现次数之积的最大值

#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
const int maxd=3e5+10;
char s[maxd];
int n,p,q,fail[maxd],cnt[maxd],len[maxd],tot,last,ch[maxd][26],pos[maxd];
//int ans[maxd];
// fail[i] 表示该串失信匹配的上一个串的cnt位置
// cnt[i] 第i个本质不同回文串的个数
// len[i] 表示该回文串的长度
// pos[i] 表示该串在原回文串中出现位置
inline int newnode(int x)
{
    len[++tot]=x;
    return tot;
}
inline int getfail(int x,int n)
{
    while(s[n-len[x]-1]!=s[n]) x=fail[x];return x;
}
inline void init()
{
    s[0]=-1,fail[0]=1,last=0;
    len[0]=0,len[1]=-1,tot=1;
}
inline void add(int i)
{
    p=getfail(last,i);
    if(!ch[p][s[i]])
    {
        q=newnode(len[p]+2);
        pos[q] = i; //该回文串出现的位置
        fail[q]=ch[getfail(fail[p],i)][s[i]]; 
        ch[p][s[i]]=q;
    }
    last=ch[p][s[i]];
    ++cnt[last];
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    while(scanf("%s",s+1)!=EOF)
    {
        init();
        for(int i=1;s[i];++i)
        {
            s[i]-='a'; add(i);
        }
        long long ans = 0ll;
        for(int i=tot;i;--i) cnt[fail[i]]+=cnt[i],ans = max(ans,(long long)len[i] * cnt[i]);
            printf("%lld\n",ans);

    }
    return 0;
}
