「Template」匈牙利算法

2019-05-04

匈牙利算法 O(n*e)

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
int maxd = 1005;
vector<int> G[maxd]; 
bool vis[maxd<<1];
int ans[maxd<<1],n,m;
bool dfs(int x)
{
	for(int v : G[x])
	{
		if(vis[v]) continue;
		vis[v] = 1;
		if(ans[v] == -1 || dfs(v))
		{
			ans[v] = x;
			return true;
		}
	}
    return false;
}
int maxmatch()
{
    int cnt = 0;
    memset(ans,-1,sizeof(ans));
    for(int i=1;i <= n;i++)
    {
        memset(vis,0,sizeof(vis));
        cnt+=dfs(i);
    }
    return cnt;
}

const ll maxd = 3005;
vector<ll> G[maxd];
bool vis[maxd<<1];
ll ans[maxd<<1],n,m;

ll dfs(ll x)
{
    for(ll v:G[x])
    {
        if(vis[v]) continue;
        vis[v] = 1;
        if(ans[v] == -1 || dfs(ans[v]))
        {
            ans[v] = x;
            return true;
        }
    }
    return false;
}
ll maxmatch()
{
    ll cnt = 0ll;
    memset(ans,-1,sizeof(ans));
    FOR(i,1,n)
    {
        memset(vis,0,sizeof(vis));
        cnt += dfs(i);
    }
    return cnt;
}
void init()
{
    FOR(i,1,2*n) G[i].clear();
}
