#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 1e5;
struct {
    int v, net;
} e[maxd << 1];
int f[maxd], eid;
void insert(int x, int y) {
    e[eid].v = y;
    e[eid].net = f[x];
    f[x] = eid++;
}
void init() {
    memset(f, -1, sizeof(f));
    eid = 0;
}
int dfn[maxd], low[maxd], tot = 0, n, s, t, ans = 1e9;
void tarjan(int x) {
    dfn[x] = low[x] = ++tot;
    for (int i = f[x]; i + 1; i = e[i].net) {
        int v = e[i].v;
        if (!dfn[v]) {
            tarjan(v);
            low[x] = min(low[v], low[x]);
            if (dfn[x] <= low[v] && x != s && x != t && dfn[t] >= dfn[v] && low[t] >= low[x])
                ans = min(ans, x);
        }
        low[x] = min(low[x], dfn[v]);
    }
}
int main() {
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    int a, b;
    init();
    scanf("%d", &n);
    while (1) {
        scanf("%d %d", &a, &b);
        if (a == 0 || b == 0)
            break;
        insert(a, b);
        insert(b, a);
    }
    scanf("%d %d", &s, &t);
    tarjan(s);
    if (ans != 1e9)
        printf("%d\n", ans);
    else
        printf("No solution");
    return 0;
}
