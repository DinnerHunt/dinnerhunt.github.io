#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <map>
using namespace std;
const int maxd = 2005;
int a[maxd],b[maxd],c[maxd],d[maxd];
int dp[maxd][maxd];
int que[maxd],l,r;
int main()
{
//    freopen("a.in","r",stdin);
//    freopen("k.out","w",stdout);
    int n,m,k;
    memset(dp,128,sizeof(dp));
    scanf("%d %d %d",&n,&m,&k);
    for(int i=1;i<=n;i++) scanf("%d %d %d %d",a+i,b+i,c+i,d+i);
    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<=c[i];j++) dp[i][j] = -1 * j * a[i];
        for(int j=0;j<=m;j++) dp[i][j] = max(dp[i][j],dp[i-1][j]);
        if(i<=k) continue;
        l = 1, r = 0;
        for(int j=0;j<=m;j++)
        {
            while( l<=r && que[l] < j-c[i]) l++;
            while( l<=r && dp[i-k-1][que[r]] + que[r] * a[i] <= dp[i-k-1][j] + j * a[i]) r--;
            que[++r] = j;
            if(l <=r ) dp[i][j] = max(dp[i][j],dp[i-k-1][que[l]] + que[l] * a[i] - j * a[i]);
        }
        l = 1,r=0;
        for(int j = m ; j >= 0 ; j--){
            while(l <= r && que[l] > j + d[i]) l++;
            while(l <= r && dp[i - k - 1][que[r]] + que[r] * b[i] <= dp[i - k - 1][j] + j * b[i]) r--;
            que[++r] = j;
            if(l <= r) dp[i][j] = max(dp[i][j] , dp[i - k - 1][que[l]] + que[l] * b[i] - j * b[i]);
        }
    }
    printf("%d",dp[n][0]);

    return 0;
}
