P4213 杜教筛(Sum)

∑i=1nϕ(i)\sum_{i=1}^{n} \phi(i)∑​i=1​n​​ϕ(i) 与 ∑i=1nμ(i)\sum_{i=1}^{n} \mu(i)∑​i=1​n​​μ(i)

#include <cstdio>
#include <algorithm>
#include <unordered_map>
#define ll long long
#define N 5000005
using namespace std;
ll mu[N],prime[N],cnt,phi[N];
bool vis[N];
unordered_map<int,long long> mpmu,mpphi;
void init()
{
	mu[1] =1; phi[1] = 1;
	for(int i=2;i<N;i++)
	{
		if(!vis[i])
		{
			prime[++cnt] = i;
			mu[i] = -1;
			phi[i] = i-1;
		}
		for(int j=1;j<=cnt && i * prime[j] < N;j++)
		{
			int p = i * prime[j]; 
			vis[p] = 1;
			if(i%prime[j] == 0)  
			{
				mu[p] = 0;
				phi[p] = phi[i] * prime[j];
				break;
			}
			phi[p] = phi[i] * (prime[j] - 1);
			mu[p] = -mu[i]; 
		}
	}
	for(int i=1;i<N;i++) mu[i] += mu[i-1];
	for(int i=1;i<N;i++) phi[i] += phi[i-1];
}
ll getsummu(int n)
{
	if(n < N) return mu[n];
	if(mpmu[n]) return mpmu[n];
	ll ans = 1ll;
	for(int l=2,r;l<=n;l = r+1)
	{
		r = n/(n/l);
		ans -= (r - l + 1) * getsummu(n/l);
	}
	return mpmu[n] = ans;
}
ll getsumphi(int n)
{
	if(n < N) return phi[n];
	if(mpphi[n]) return mpphi[n];
	ll ans = 1ll * (1+n)*n/2;
	for(int l=2,r;l<=n;l = r+1)
	{
		r = n/(n/l);
		ans -= (r - l + 1) * getsumphi(n/l);
	}
	return mpphi[n] = ans;
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    init();
    int n,x,y;scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
    	scanf("%d",&x);
    	printf("%lld %lld\n",getsumphi(x),getsummu(x));
    }
    return 0;
}
