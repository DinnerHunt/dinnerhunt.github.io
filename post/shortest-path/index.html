zkw线段树优化-dijkstra

#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 2500;		//点数
const int maxn = 10000;		//边数
const int inf = 1e9;
struct node
{
	int v,w,next;
}e[maxn*2];
int f[maxd],eid;
void init()
{
	memset(f,-1,sizeof(f));
	eid = 0;
}
void insert(int a,int b,int c)
{
	e[eid].v = b;
	e[eid].w = c;
	e[eid].next = f[a];
	f[a] = eid++;
}
struct
{
	int minx[maxd*4],v[maxd*4];
	int n,len;
	void push_up(int x)
	{
		minx[x] = min(minx[x<<1],minx[x<<1|1]);
		v[x] = minx[x<<1] < minx[x<<1|1] ? v[x<<1] : v[x<<1|1];
	}
	void build()
	{
		for(len=1;len<=n+1;len<<=1);
		for(int i=len;i<=(len<<1)-1;i++)
		{
			minx[i] = inf;
			v[i] = i-len;
		}
		for(int i=len-1;i;i--) push_up(i);
	}
	void updata(int x,int k)
	{
		for(minx[x+=len]=k,x>>=1;x;x>>=1) 
		{
			//printf("%d ",x);
			push_up(x);
		}
	}
}tree;
int dist[maxd],n,m,ts,te,a,b,c;
void dijkstra()
{
	//ts 起始点，te终点，n总点数
	memset(dist,0x3f,sizeof(dist));
	tree.n = n;
	tree.build();
	dist[ts] = 0;
	tree.updata(ts,0);
	while(tree.minx[1]!=inf)
	{
		int v = tree.v[1];
		tree.updata(v,inf);
		for(int i=f[v];i+1;i=e[i].next)
		{
			int s = e[i].v,w = e[i].w;
			if(dist[v]+w < dist[s])
			{
				dist[s] = dist[v] + w;
				tree.updata(s,dist[s]);
			}
		}
	}
	printf("%d",dist[te]);
}
int main()
{
	// freopen("a.in","r",stdin);
	// freopen("k.out","w",stdout);
	init();
	scanf("%d %d %d %d",&n,&m,&ts,&te);
	for(int i=1;i<=m;i++)
	{
		scanf("%d %d %d",&a,&b,&c);
		insert(a,b,c);
		insert(b,a,c);
	}
	dijkstra();
	return 0;
}
