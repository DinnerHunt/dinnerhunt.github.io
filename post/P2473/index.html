#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 1<<15;
double dp[100][maxd];
int arr[16],val[16];
int n,k;
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%d %d",&k,&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&val[i]);
        int x;
        while(1)
        {
            scanf("%d",&x);
            if(!x) break;
            arr[i] |= (1<<(x-1));
        }
    }
    for(int i=k;i>=1;i--)
        for(int j=0;j<(1<<n);j++)
            for(int l=1;l<=n;l++)
            {
                if((j&arr[l])==arr[l])
                    dp[i][j] += max(dp[i+1][j|(1<<(l-1))]+val[l],dp[i+1][j])*1.0/n;
                else dp[i][j] += dp[i+1][j]*1.0/n;
            }
    printf("%lf",dp[1][0]);
    return 0;
}
