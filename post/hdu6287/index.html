「HDU 6287」口算训练

2020-03-19 主席树

小Q非常喜欢数学，但是他的口算能力非常弱。因此他找到了小T，给了小T一个长度为nnn的正整数序列a1,a2,...,ana_1,a_2,...,a_na1​,a2​,...,an​，要求小T抛出mmm个问题以训练他的口算能力。

每个问题给出三个正整数l,r,dl,r,dl,r,d，小Q需要通过口算快速判断al×al+1×...×ar−1×ara_l\times a_{l+1}\times ...\times a_{r-1}\times a_ral​×al+1​×...×ar−1​×ar​是不是ddd的倍数。

小Q迅速地回答了出来，但是小T并不知道正确答案是什么，请写一个程序帮助小T计算这些问题的正确答案。

题解 ❓

主席树
线段树的节点存该数所包含该质因子的数量
因为数值范围在1e51e51e5，因此一个数不会包含大于10个的不同质因子
因此对于一个数，我们最多进行10次修改操作

代码

const ll N = 1e5+10;
const ll mod = 998244353;
ll id[N];
struct
{
    bool vis[N];
    ll p[N],cnt;
    void init()
    {
        cnt = 0;memset(vis,0,sizeof(vis));
        FOR(i, 2, N - 1)
        {
            if(!vis[i]) p[++cnt] = i,id[i] = cnt;
            for(ll j=1;j<=cnt && i * p[j] < N; j++)
            {
                vis[i*p[j]] = 1;
                if(!(i%p[j])) break;
            }
        }
    }
}O_O;
ll tt[N],t[N*100],ls[N*100],rs[N*100],tot;
void insert(ll &rt,ll last,ll l,ll r,ll v,ll num)
{
    rt = ++tot;
//    PLN4(rt,l,r,v);
    ls[rt] = ls[last],rs[rt] = rs[last];
    t[rt] = t[last]+num;
    if(l == r) return;
    if(v<=mid) insert(ls[rt],ls[last],l,mid,v,num);
    else  insert(rs[rt],rs[last],mid+1,r,v,num);
}
ll query(ll &rt,ll last,ll l,ll r,ll v,ll num)
{
    if(l == r) return t[rt] - t[last];
    if(t[rt] - t[last] < num) return 0ll;
    if(v<=mid) return query(ls[rt],ls[last],l,mid,v,num);
    else return query(rs[rt],rs[last],mid+1,r,v,num);
}
int main()
{
//    __IN;__OUT
    O_O.init();
    ll T;RLL(T);
    while(T--)
    {
        ll n,m;RLL2(n,m);
        tt[0] = 0,tot=0;
        ll x,y,z;
        FOR(i,1,n)
        {
            RLL(x); tt[i] = tt[i-1];
            for(ll j=1;O_O.p[j] * O_O.p[j] <= x && j <= O_O.cnt;j++)
            {
                ll num = 0;
                while(x%O_O.p[j] == 0) num++,x /= O_O.p[j];
                if(num) insert(tt[i],tt[i],1,O_O.cnt,id[O_O.p[j]],num);
            }
            if(x>1) insert(tt[i],tt[i],1,O_O.cnt,id[x],1);
        }
        FOR(i,1,m)
        {
            RLL3(x,y,z); ll flag = 1;
            for(ll j=1;O_O.p[j] * O_O.p[j] <= z;j++)
            {
                ll num = 0;
                while(z%O_O.p[j] == 0) num++,z /= O_O.p[j];
                if(num)
                {
                    flag &= (query(tt[y],tt[x-1],1,O_O.cnt,id[O_O.p[j]],num) >= num);
//                    PLN3(O_O.p[j],num,query(tt[y],tt[x-1],1,1e5,O_O.p[j],num));
                }
            }
            if(z>1)
            {
                flag &= (query(tt[y],tt[x-1],1,O_O.cnt,id[z],1) >=1);
//                PLN3(z,1,query(tt[y],tt[x-1],1,1e5,z,1));
            }
            if(flag) PS("Yes");
            else PS("No");
        }

    }
    return 0;
}
