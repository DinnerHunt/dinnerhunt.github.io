#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxd = 1e7+10;
int m,n,ans,k,t;
int mu[maxd], prime[maxd],summu[maxd],cnt,f[maxd];
bool vis[maxd];
void init() 
{
    mu[1] = 1;
    for(int i=2;i<maxd;i++)
    {
        if(!vis[i]) prime[++cnt] = i, mu[i] = -1;
        for(int j=1;j<=cnt && i*prime[j] <= maxd;j++)
        {
            vis[i*prime[j]] = 1;
            if(i%prime[j] == 0) break;
            mu[i*prime[j]] = -mu[i]; 
        }
    }
    for(int i=1;i<=cnt;i++)
        for(int j=1;prime[i]*j<maxd;j++)
            f[j*prime[i]] += mu[j];
		//å‰ç¼€å’Œ
    for(int i=1;i<maxd;i++) summu[i] = summu[i-1] + f[i];
}
long long sum(int m,int n)  //æ•°è®ºåˆ†å—
{
    long long t = min(n, m), ans = 0;
    for (int i = 1, j = 1; i <= t; i = j + 1) 
    {
        j = min(n / ( n / i), m / (m / i));
        ans += (summu[j] - summu[i - 1]) * (long long)(n / i) * (m / j);
    }
    return ans;
}
int main()
{

    {

    }
    return 0;
}
