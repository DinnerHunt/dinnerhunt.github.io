「Template」ST表

2019-05-04

O(nlogn)O(n log n)O(nlogn) 预处理
O(1)O(1)O(1) 查询

struct
{
    ll mxp[maxd][20],mip[maxd][20];
    void init(ll *a,ll n)
    {
        FOR(i,1,n) mxp[i][0] = mip[i][0] = a[i];
        for(int j=1;(1<<j)-1<=n;j++)
            for(int i=1;i+(1<<j)-1<=n;i++)
            {
                mxp[i][j] = max(mxp[i][j-1],mxp[i+(1<<(j-1))][j-1]);
                mip[i][j] = min(mip[i][j-1],mip[i+(1<<(j-1))][j-1]);
            }
    }
    ll mx(ll l, ll r)
    {
        ll k = (int)(log((double)(r-l+1))/log(2.0));
        return max(mxp[l][k],mxp[r-(1<<k)+1][k]);
    }
    ll mi(ll l, ll r)
    {
        ll k = (int)(log((double)(r-l+1))/log(2.0));
        return min(mip[l][k],mip[r-(1<<k)+1][k]);
    }
}st;
