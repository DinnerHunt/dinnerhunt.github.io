O(nlogn)O(n log n)O(nlogn) 预处理
O(1)O(1)O(1) 查询

#include<cstdio>
#include<cmath>
#include <algorithm>
using namespace std;
const int maxd = 1e5+10;
int map[maxd][20],arr[maxd],n,a,b,m;
void init()
{
    for(int i=1;i<=n;i++) map[i][0]=arr[i];
    for(int j=1;(1<<j)-1<=n;j++)
        for(int i=1;i+(1<<j) - 1<=n;i++)
            map[i][j] = max(map[i][j-1],map[i+(1<<(j-1))][j-1]);
}
int query(int l,int r)
{
    int k = (int)(log((double)(r-l+1))/log(2.0));
    return max(map[l][k],map[r-(1<<k)+1][k]);
}
int main()
{
    // freopen("a.in","r",stdin);
    // freopen("k.out","w",stdout);
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%d",&arr[i]);init();
    for(int i=1;i<=m;i++)
    {
        scanf("%d %d",&a,&b);
        printf("%d\n",query(a,b));
    }
    return 0;
}
