求字符串str的所有后缀，与字符串T的最大公共前缀


int net[maxd],extend[maxd];
// net[i],以字符串T的第i项为起点的后缀与字符串T的最大公共前缀
// extend[i], 以字符串str第i项为起点的后缀与字符串T的最大公共前缀
char T[maxd],str[maxd];
void getnet()
{
    net[0] = strlen(T);
    int p = 0,k=1;
    while(T[p+1]&&T[p] == T[p+1]) p++;net[k] = p;
    for(int i=2;T[i];i++)
    {
        p = k + net[k] - 1;
        int l = net[i-k];
        if(i+l <= p) net[i] = l;
        else
        {
            int j = p - i +1; // 当前匹配到的位置
            j = max(j,0);
            while(T[i+j] && T[i+j] == T[j]) j++;
            net[i] = j;
            k = i;
        }
    }
}
void exkmp()
{
    int p = 0,k = 0;
    while(T[p] && str[p] && T[p] == str[p]) p++;
    extend[k] = p;
    for(int i=1;str[i];i++)
    {
        p = k + extend[k] - 1;
        int l = net[i-k];
        if(i+l <= p ) extend[i] = l;
        else
        {
            int j = p - i +1; j = max(j,0);
            while(T[j]&&str[i+j]&&T[j] == str[i+j]) j++;
            extend[i] = j,k = i;
        }
    }
}
