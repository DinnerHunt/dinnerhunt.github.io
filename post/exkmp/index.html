{
	net[0] = strlen(T);
	int p = 0; 
	while(T[p+1]&&T[p] == T[p+1]) p++;
	int k = 1;
	net[k] = p;
	for(int i=2;T[i];i++)
	{
		p = k + net[k] - 1;
		int l = net[i-k];
		if(i+l <= p) net[i] = l;
		else 
		{
			int j = p - i +1; // 当前匹配到的位置
			j = max(j,0);
			while(T[i+j] && T[i+j] == T[j]) j++;
			net[i] = j;
			k = i;
		}
	}
}
